## 二、ReAct Agent 完整实现

### 基础框架
```python
from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import Any, Dict, List, Callable

@dataclass
class Tool:
    """工具定义"""
    name: str
    description: str
    func: Callable
    parameters: Dict[str, Any]

class SimpleAgent:
    """基于 ReAct 模式的 Agent"""
    
    def __init__(self, model, tools: List[Tool], max_steps=5):
        self.model = model
        self.tools = {t.name: t for t in tools}
        self.max_steps = max_steps
        self.history: List[Dict] = []
    
    def _build_system_prompt(self) -> str:
        tools_desc = "\n".join([
            f"- {t.name}: {t.description}\n  Parameters: {t.parameters}"
            for t in self.tools.values()
        ])
        
        return f"""You are an AI agent that can use tools to complete tasks.

## Available Tools
{tools_desc}

## Response Format
For each step, respond in JSON:
{{"thought": "your reasoning", "action": "tool_name", "action_input": {{...}}}}
Or when the task is complete:
{{"thought": "task done", "final_answer": "your answer"}}
"""
    
    def run(self, task: str) -> str:
        """执行任务"""
        messages = [
            {"role": "system", "content": self._build_system_prompt()},
            {"role": "user", "content": task}
        ]
        
        for step in range(self.max_steps):
            response = self.model.chat(messages)
            
            # 解析响应
            try:
                parsed = json.loads(response)
            except json.JSONDecodeError:
                messages.append({"role": "user", "content": "Please respond in valid JSON format."})
                continue
            
            # 检查是否完成
            if "final_answer" in parsed:
                return parsed["final_answer"]
            
            # 执行工具
            action = parsed.get("action")
            action_input = parsed.get("action_input", {})
            
            if action not in self.tools:
                observation = f"Error: Tool '{action}' not found. Available: {list(self.tools.keys())}"
            else:
                try:
                    result = self.tools[action].func(**action_input)
                    observation = f"Tool '{action}' result: {result}"
                except Exception as e:
                    observation = f"Error executing '{action}': {str(e)}"
            
            messages.append({"role": "assistant", "content": response})
            messages.append({"role": "user", "content": f"Observation: {observation}\n\nWhat's your next step?"})
        
        return "Task exceeded maximum steps."
```

## 三、Plan-and-Execute Agent

### 先规划再执行
```python
class PlanAndExecuteAgent:
    """Plan-and-Execute 模式"""
    
    def run(self, task: str) -> str:
        # Step 1: 制定计划
        plan = self._create_plan(task)
        print(f"📋 Plan: {plan}")
        
        # Step 2: 逐步执行
        results = []
        for step in plan:
            print(f"▶ Executing: {step}")
            result = self._execute_step(step)
            results.append(result)
        
        # Step 3: 汇总结果
        return self._synthesize(task, plan, results)
    
    def _create_plan(self, task: str) -> List[str]:
        prompt = f"""Break down the following task into detailed steps:
Task: {task}

Output a JSON list of steps:
["Step 1: ...", "Step 2: ...", ...]"""
        
        response = self.model.generate(prompt)
        return json.loads(response)
```

## 四、Multi-Agent 系统

### 多 Agent 协作
```python
class MultiAgentSystem:
    """多 Agent 协作系统"""
    
    def __init__(self):
        self.agents = {
            "researcher": Agent("researcher", "Research and gather information"),
            "analyst": Agent("analyst", "Analyze data and draw conclusions"),
            "writer": Agent("writer", "Write clear and engaging content"),
            "reviewer": Agent("reviewer", "Review and provide feedback"),
        }
    
    def run_pipeline(self, task: str) -> str:
        """流水线模式：Agent 串行执行"""
        research_result = self.agents["researcher"].run(task)
        analysis_result = self.agents["analyst"].run(f"{task}\n\nResearch: {research_result}")
        draft = self.agents["writer"].run(f"{task}\n\nAnalysis: {analysis_result}")
        feedback = self.agents["reviewer"].run(draft)
        final = self.agents["writer"].run(f"Revise based on feedback:\n\nDraft: {draft}\nFeedback: {feedback}")
        return final
    
    def run_chat(self, task: str) -> str:
        """对话模式：Agent 互相讨论"""
        conversation = [{"from": "user", "content": task}]
        
        for round_num in range(3):
            for name, agent in self.agents.items():
                response = agent.respond(conversation)
                conversation.append({"from": name, "content": response})
                print(f"💬 {name}: {response[:100]}...")
        
        return self._extract_conclusion(conversation)
```

### Multi-Agent 示例：代码 Review
```python
def code_review_multi_agent(code: str):
    """多 Agent 代码审查"""
    
    coder = Agent("coder", "Write and fix code")
    security_expert = Agent("security_expert", "Find security vulnerabilities")
    reviewer = Agent("reviewer", "Review code quality and style")
    
    # 并行审查
    import concurrent.futures
    with concurrent.futures.ThreadPoolExecutor() as executor:
        security_future = executor.submit(
            security_expert.run, f"Find security issues in:\n{code}"
        )
        review_future = executor.submit(
            reviewer.run, f"Review code quality:\n{code}"
        )
        
        security_report = security_future.result()
        review_report = review_future.result()
    
    # 汇总修复
    fix_prompt = f"""Fix the following code based on review feedback:

Code:
{code}

Security Issues:
{security_report}

Code Quality Issues:
{review_report}

Provide the fixed code:"""
    
    fixed_code = coder.run(fix_prompt)
    return fixed_code
```

### LangGraph 示例
```python
from langgraph.graph import StateGraph, END
from typing import TypedDict, Annotated
import operator

class AgentState(TypedDict):
    task: str
    messages: Annotated[list, operator.add]
    next_step: str
    final_answer: str

def should_continue(state: AgentState) -> str:
    """判断是否需要继续循环"""
    if state["next_step"] == "finish":
        return END
    return state["next_step"]

# 构建图
workflow = StateGraph(AgentState)

workflow.add_node("agent", agent_node)
workflow.add_node("tools", tool_node)

workflow.set_entry_point("agent")
workflow.add_conditional_edges("agent", should_continue, {
    "tools": "tools",
    END: END
})
workflow.add_edge("tools", "agent")

graph = workflow.compile()
result = graph.invoke({"task": "Find the latest AI news"})
```
