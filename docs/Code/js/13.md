# ES6+ 常用新特性

ES6（ES2015）是 JavaScript 历史上最大的一次版本更新，此后每年发布一个版本。本章汇总日常开发中最常用的 ES6+ 特性。

## 解构赋值（Destructuring）

```javascript
// 数组解构
const colors = ["red", "green", "blue"];
const [first, second, third] = colors;
const [a, , c] = colors;               // "red", "blue" — 跳过中间元素
const [head, ...rest] = colors;        // "red", ["green", "blue"] — 剩余元素

// 带默认值
const [x = 1, y = 2] = [10];           // x=10, y=2

// 交换变量（无需临时变量）
let m = 1, n = 2;
[m, n] = [n, m];                       // m=2, n=1

// 对象解构
const user = { name: "Alice", age: 25, address: { city: "Beijing" } };
const { name, age } = user;            // 提取同名属性
const { name: userName, age: userAge } = user; // 重命名

// 嵌套解构
const { address: { city } } = user;    // "Beijing"

// 带默认值
const { role = "user" } = user;        // 属性不存在，使用默认值

// 函数参数解构
function printInfo({ name, age = 18 }) {
  console.log(`${name}, ${age} 岁`);
}
printInfo(user); // "Alice, 25 岁"
printInfo({ name: "Bob" }); // "Bob, 18 岁"
```

## 扩展运算符与剩余参数（Spread / Rest）

`...` 在不同的上下文中有不同含义——展开（spread）或收集（rest）。

```javascript
// Spread — 展开可迭代对象
const arr1 = [1, 2, 3];
const arr2 = [4, 5, 6];
const merged = [...arr1, ...arr2];      // [1, 2, 3, 4, 5, 6]

const obj1 = { x: 1, y: 2 };
const obj2 = { y: 3, z: 4 };
const mergedObj = { ...obj1, ...obj2 }; // { x: 1, y: 3, z: 4 } — 后者覆盖前者

// 展开作为函数参数
Math.max(...arr1);                      // 3 — 替代 Math.max.apply(null, arr1)

// Rest — 收集剩余参数
function sum(prefix, ...numbers) {
  return `${prefix}: ${numbers.reduce((a, b) => a + b, 0)}`;
}
sum("总计", 1, 2, 3, 4); // "总计: 10"

// Rest 在解构中收集剩余属性
const { name, ...restProps } = user;    // name="Alice", restProps={ age: 25, address: {city: "Beijing"} }
```

## 默认参数

```javascript
// 基础默认值
function greet(name = "Guest") {
  return `Hello, ${name}`;
}
greet();        // "Hello, Guest"
greet("Alice"); // "Hello, Alice"

// 默认值可以使用之前的参数
function createPoint(x = 0, y = x) {
  return { x, y };
}
createPoint(5); // { x: 5, y: 5 }

// 默认值可以是函数调用（惰性求值）
function getDefaultId() {
  return Date.now();
}
function create(id = getDefaultId()) {
  return { id };
}
```

## Map 与 Set

### Map — 键值对集合，键可以是任意类型

```javascript
const map = new Map();

// 设置与获取
map.set("name", "Alice");
map.set(1, "number key");
map.set({ id: 1 }, "object key");
map.set(true, "boolean key");

map.get("name");         // "Alice"
map.has("name");         // true
map.size;                // 4

// 遍历
for (const [key, value] of map) {
  console.log(key, value);
}
map.forEach((value, key) => console.log(key, value));

// 与 Object 的区别
const obj = {};
map.set(obj, "value for object");
map.get(obj); // "value for object" — Object 键会被转为字符串，Map 不会

// WeakMap — 键必须是对象，不影响垃圾回收
const weakMap = new WeakMap();
let el = document.querySelector("#app");
weakMap.set(el, { clicked: 0 });
// 当 el 被移除且无其他引用时，WeakMap 中的条目会被自动回收
```

### Set — 唯一值的集合

```javascript
const set = new Set([1, 2, 3, 3, 4, 4, 5]);
console.log(set);      // Set { 1, 2, 3, 4, 5 } — 自动去重
console.log(set.size); // 5

// 操作
set.add(6);
set.delete(2);
set.has(3);            // true
set.clear();

// 遍历
for (const value of set) { console.log(value); }

// Set 运算（并集、交集、差集）
const a = new Set([1, 2, 3]);
const b = new Set([2, 3, 4]);

const union = new Set([...a, ...b]);        // {1, 2, 3, 4}
const intersect = new Set([...a].filter(x => b.has(x))); // {2, 3}
const diff = new Set([...a].filter(x => !b.has(x))); // {1}

// WeakSet — 只能存对象，不影响 GC
const weakSet = new WeakSet();
weakSet.add(el);
```

## Symbol

Symbol 是 ES6 引入的原始类型，用于创建**全局唯一**的值。

```javascript
const sym1 = Symbol("desc");
const sym2 = Symbol("desc");
console.log(sym1 === sym2); // false — 每个 Symbol 都是唯一的

// 作为对象属性键（不会被常规方法遍历）
const id = Symbol("id");
const user = {
  name: "Alice",
  [id]: 12345,
};

console.log(user[id]); // 12345
console.log(Object.keys(user));          // ["name"] — Symbol 属性被隐藏
console.log(Object.getOwnPropertySymbols(user)); // [Symbol(id)]

// Symbol.for — 全局符号注册表
const globalSym1 = Symbol.for("app.key");
const globalSym2 = Symbol.for("app.key");
console.log(globalSym1 === globalSym2); // true — 同一 key 返回相同 Symbol

// 内置 Well-known Symbols
class MyArray {
  [Symbol.iterator]() { /* 自定义迭代器 */ }
  [Symbol.toPrimitive](hint) { return 42; } // 类型转换
  [Symbol.toStringTag] = "MySpecialClass"; // Object.prototype.toString 返回
}
```

## 模块化 import / export

ES Modules（ESM）是 JavaScript 官方的模块系统。

```javascript
// ========== module: math.js ==========
// 命名导出
export const PI = 3.14159;

export function add(a, b) {
  return a + b;
}

export function multiply(a, b) {
  return a * b;
}

// 默认导出（每个模块只能有一个）
export default function subtract(a, b) {
  return a - b;
}

// 批量导出
const utils = { log: console.log };
export { utils };

// ========== module: app.js ==========
// 命名导入
import { PI, add } from "./math.js";

// 别名导入
import { multiply as mul } from "./math.js";

// 默认导入
import subtract from "./math.js";

// 命名空间导入（导入所有命名导出为一个对象）
import * as math from "./math.js";
console.log(math.PI, math.add(1, 2));

// 同时导入默认和命名导出
import subtract, { add, PI } from "./math.js";

// 动态导入 — 返回 Promise，支持按需加载
async function loadModule() {
  const { add } = await import("./math.js");
  console.log(add(1, 2));
}

// 仅执行模块代码（不导入任何内容）
import "./init.js";
```

**HTML 中使用 ESM**：

```html
<!-- type="module" 会延迟执行（defer）、默认严格模式、独立作用域 -->
<script type="module" src="app.js"></script>
<script type="module">
  import { add } from "./math.js";
  console.log(add(1, 2));
</script>
```

## 其他实用特性

```javascript
// 指数运算符（ES2016）
2 ** 10; // 1024

// Object.entries / Object.values / Object.fromEntries（ES2017）
Object.entries({ a: 1, b: 2 }); // [["a",1],["b",2]]
Object.values({ a: 1, b: 2 });  // [1, 2]
Object.fromEntries([["a", 1], ["b", 2]]); // { a: 1, b: 2 }

// String padding（ES2017）
"5".padStart(3, "0");  // "005"
"5".padEnd(3, "0");    // "500"

// flat / flatMap（ES2019）
[1, [2, [3]]].flat(2);         // [1, 2, 3]
[1, 2, 3].flatMap(x => [x, x * 2]); // [1, 2, 2, 4, 3, 6]

// 可选链 ?. (ES2020)
user?.address?.city;

// 空值合并 ?? (ES2020)
const count = value ?? 0;

// 逻辑赋值 (ES2021)
config.timeout ??= 3000;

// .at() 支持负索引 (ES2022)
const arr = [1, 2, 3];
arr.at(-1); // 3 — 等效于 arr[arr.length - 1]

// 顶层 await (ES2022, 仅 ESM)
// const data = await fetch("/api").then(r => r.json());
// export { data };
```

## 总结

- **解构**是日常最常用的特性之一，大幅减少 `const x = obj.x` 这类冗余代码
- **Spread** 用于展开、浅拷贝、合并；**Rest** 用于收集剩余项
- **Map** 的键可以是任意类型，**Set** 天然去重
- **Symbol** 适合创建私有属性键或避免命名冲突
- **ES Modules** 是现代 JS 的标准模块方案，优于 CommonJS
- 每年新特性通过 **Babel** 或 **TypeScript** 可以放心使用
