# 函数

函数是 JavaScript 的核心构建块。JS 中的函数是"一等公民"（first-class citizen），可以赋值给变量、作为参数传递、从函数返回。

## 函数声明 vs 函数表达式

```javascript
// 函数声明（Function Declaration）— 存在函数提升，可在定义前调用
console.log(add(1, 2)); // 3

function add(a, b) {
  return a + b;
}

// 函数表达式（Function Expression）— 无函数提升
// console.log(subtract(5, 2)); // TypeError: subtract is not a function

const subtract = function (a, b) {
  return a - b;
};

console.log(subtract(5, 2)); // 3
```

**区别**：函数声明整体提升到作用域顶部，可以在声明前调用。函数表达式只提升变量名（值为 `undefined`），赋值前调用会报错。

## 箭头函数（Arrow Functions）

ES6 引入的更简洁的函数写法，且**不绑定自己的 `this`**。

```javascript
// 基本语法
const add = (a, b) => a + b;          // 单行隐式返回
const double = n => n * 2;            // 单个参数可省略括号
const greet = () => "Hello";          // 无参数用空括号
const createUser = (name, age) => ({  // 返回对象需要括号包裹
  name,
  age
});

// 多行函数体需要大括号和显式 return
const process = (data) => {
  const validated = validate(data);
  return transform(validated);
};
```

**箭头函数与普通函数的区别**：

```javascript
// 1. 箭头函数没有自己的 this，继承外层作用域
const obj = {
  name: "Alice",
  regularFunc: function () {
    console.log(this.name); // "Alice" — this 指向 obj
  },
  arrowFunc: () => {
    console.log(this.name); // undefined — this 继承外层（全局/window）
  }
};

// 2. 箭头函数不能用作构造函数
// const Person = (name) => { this.name = name; };
// new Person("Bob"); // TypeError

// 3. 箭头函数没有 arguments 对象
const regular = function () { console.log(arguments); };
const arrow = () => { console.log(arguments); };
regular(1, 2, 3); // [1, 2, 3]
// arrow(1, 2, 3); // ReferenceError: arguments is not defined

// 4. 箭头函数没有 prototype 属性
console.log(regular.prototype); // { constructor: f }
console.log(arrow.prototype);   // undefined
```

## 参数处理

```javascript
// 默认参数（Default Parameters）
function greet(name = "Guest", greeting = "Hello") {
  return `${greeting}, ${name}!`;
}
console.log(greet());             // "Hello, Guest!"
console.log(greet("Alice"));      // "Hello, Alice!"
console.log(greet(undefined, "Hi")); // "Hi, Guest!" — undefined 触发默认值

// 剩余参数（Rest Parameters）
function sum(multiplier, ...numbers) {
  return numbers.reduce((acc, n) => acc + n, 0) * multiplier;
}
console.log(sum(2, 1, 2, 3, 4)); // 20 — 1+2+3+4=10, 10*2=20

// arguments 对象（普通函数才有）
function oldStyle() {
  console.log(Array.from(arguments).filter(x => x > 2));
}
oldStyle(1, 2, 3, 4); // [3, 4]
```

## 闭包（Closure）

闭包是指函数能够访问其外部作用域变量的能力，即使外部函数已经返回。

```javascript
function createCounter(initial = 0) {
  let count = initial; // 私有变量

  return {
    increment: () => ++count,
    decrement: () => --count,
    getCount: () => count,
  };
}

const counter = createCounter(10);
console.log(counter.increment()); // 11
console.log(counter.increment()); // 12
console.log(counter.decrement()); // 11
console.log(counter.getCount());  // 11

// 实际应用：防抖（debounce）
function debounce(fn, delay) {
  let timer = null;
  return function (...args) {
    clearTimeout(timer);
    timer = setTimeout(() => fn.apply(this, args), delay);
  };
}

// 实际应用：截流（throttle）
function throttle(fn, interval) {
  let lastTime = 0;
  return function (...args) {
    const now = Date.now();
    if (now - lastTime >= interval) {
      lastTime = now;
      fn.apply(this, args);
    }
  };
}
```

## 立即执行函数表达式（IIFE）

IIFE（Immediately Invoked Function Expression）立即执行，用于创建独立作用域：

```javascript
// 经典 IIFE
(function () {
  const privateVar = "I'm private";
  console.log(privateVar);
})();

// 箭头函数 IIFE
(() => {
  const anotherPrivate = "also private";
  console.log(anotherPrivate);
})();

// 异步 IIFE（常见于顶层 await 不可用的场景）
(async () => {
  const data = await fetch("/api/users");
  const users = await data.json();
  console.log(users);
})();
```

## 递归（Recursion）

函数调用自身，适合处理树形结构、阶乘、斐波那契等问题：

```javascript
// 阶乘
function factorial(n) {
  if (n <= 1) return 1;
  return n * factorial(n - 1);
}
console.log(factorial(5)); // 120

// 深度遍历嵌套数组
function flattenDeep(arr) {
  return arr.reduce((acc, item) =>
    Array.isArray(item)
      ? acc.concat(flattenDeep(item))
      : acc.concat(item),
    []
  );
}
console.log(flattenDeep([1, [2, [3, [4]]]])); // [1, 2, 3, 4]

// 尾递归优化（注意：多数引擎尚未广泛实现）
function factorialTail(n, acc = 1) {
  if (n <= 1) return acc;
  return factorialTail(n - 1, n * acc);
}
```

## 纯函数与副作用

```javascript
// 纯函数 — 相同输入永远得到相同输出，无副作用
function add(a, b) {
  return a + b;
}

// 副作用函数 — 修改外部状态、I/O、DOM 操作等
let total = 0;
function addToTotal(amount) {
  total += amount; // 副作用：修改外部变量
  console.log(total); // 副作用：I/O
}
```

## 总结

- 默认使用箭头函数（简洁、无 this 绑定问题），需要动态 `this` 时用普通函数
- 闭包用于创建私有变量、防抖截流、柯里化等场景
- 剩余参数 `...args` 替代 `arguments` 对象
- 递归适合处理嵌套/树形结构，注意设置终止条件避免栈溢出
- IIFE 在现代模块化开发中使用减少，但在脚本注入等场景仍有价值
