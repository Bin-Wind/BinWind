# 输入验证与数据清洗 (Joi / Zod)

不做验证的 API 是定时炸弹：脏数据入库、SQL 注入风险、糟糕的错误提示。本文介绍 Joi 和 Zod 两套最流行的 Node.js 验证库。

## 为什么需要验证

```js
// ❌ 不验证 — 危险
app.post('/api/users', (req, res) => {
  const { name, email, age } = req.body;
  // 如果 email 不是合法格式？age 是负数？name 是空字符串？
  db.createUser({ name, email, age });
  res.json({ ok: true });
});

// ✅ 验证后处理
app.post('/api/users', validate(createUserSchema), (req, res) => {
  // req.body 已经是安全的数据
  db.createUser(req.body);
  res.json({ ok: true });
});
```

## Joi 入门

### 安装

```bash
npm install joi
```

### 定义 Schema

```js
const Joi = require('joi');

const createUserSchema = Joi.object({
  username: Joi.string()
    .alphanum()
    .min(3)
    .max(30)
    .required()
    .messages({
      'string.min': '用户名至少 3 个字符',
      'string.max': '用户名最多 30 个字符',
      'any.required': '用户名为必填项',
    }),

  email: Joi.string()
    .email()
    .required()
    .messages({
      'string.email': '请输入有效的邮箱地址',
    }),

  password: Joi.string()
    .min(8)
    .max(128)
    .pattern(/^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)/)
    .required()
    .messages({
      'string.pattern.base': '密码必须包含大小写字母和数字',
    }),

  age: Joi.number()
    .integer()
    .min(0)
    .max(150)
    .optional(),

  role: Joi.string()
    .valid('user', 'admin', 'moderator')
    .default('user'),

  tags: Joi.array()
    .items(Joi.string().max(20))
    .max(10),

  address: Joi.object({
    city: Joi.string().required(),
    street: Joi.string().required(),
    zipCode: Joi.string().pattern(/^\d{6}$/),
  }).optional(),
});
```

### 验证数据

```js
const { error, value } = createUserSchema.validate({
  username: 'ab',             // 太短
  email: 'not-an-email',      // 格式错误
  password: '12345678',       // 缺少大小写组合
});

if (error) {
  console.log(error.details);
  // [
  //   { message: '用户名至少 3 个字符', path: ['username'] },
  //   { message: '请输入有效的邮箱地址', path: ['email'] },
  //   { message: '密码必须包含大小写字母和数字', path: ['password'] },
  // ]
}
```

`validate()` 返回的 `value` 中包含了默认值和类型转换后的结果，是使用 `value` 而不是原始 `req.body`。

### Express 中间件封装

```js
function validate(schema) {
  return (req, res, next) => {
    const { error, value } = schema.validate(req.body, {
      abortEarly: false,   // 返回全部错误，而非遇到第一个就停止
      stripUnknown: true,  // 移除 schema 未定义的字段
    });

    if (error) {
      const messages = error.details.map((d) => ({
        field: d.path.join('.'),
        message: d.message,
      }));
      return res.status(400).json({ error: '请求参数验证失败', details: messages });
    }

    req.body = value; // 替换为清洗后的数据
    next();
  };
}

// 使用
app.post('/api/users', validate(createUserSchema), (req, res) => {
  // req.body 已通过验证
  const user = createUser(req.body);
  res.status(201).json(user);
});
```

## Zod 入门

Zod 是 Next.js / tRPC 生态中广泛使用的验证库，原生支持 TypeScript 类型推断。

### 安装

```bash
npm install zod
```

### 定义 Schema

```js
const { z } = require('zod');

const createUserSchema = z.object({
  username: z.string()
    .min(3, '用户名至少 3 个字符')
    .max(30, '用户名最多 30 个字符')
    .regex(/^[a-zA-Z0-9_]+$/, '用户名只能包含字母、数字和下划线'),

  email: z.string()
    .email('请输入有效的邮箱地址'),

  password: z.string()
    .min(8, '密码至少 8 位')
    .regex(/^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)/, '密码必须包含大小写字母和数字'),

  age: z.number()
    .int()
    .min(0)
    .max(150)
    .optional(),

  role: z.enum(['user', 'admin', 'moderator']).default('user'),

  tags: z.array(z.string().max(20)).max(10).optional(),

  address: z.object({
    city: z.string(),
    street: z.string(),
    zipCode: z.string().regex(/^\d{6}$/, '邮编格式错误'),
  }).optional(),
});
```

### 验证：parse vs safeParse

```js
// parse() — 验证失败时抛出 ZodError
try {
  const data = createUserSchema.parse(req.body);
  console.log('验证通过:', data);
} catch (err) {
  if (err instanceof z.ZodError) {
    console.log(err.errors);
  }
}

// safeParse() — 始终返回 { success, data } 或 { success, error }
const result = createUserSchema.safeParse(req.body);

if (!result.success) {
  console.log(result.error.errors);
  // [{ code: 'too_small', minimum: 3, path: ['username'], message: '...' }]
} else {
  console.log(result.data); // 验证通过的数据
}
```

**推荐使用 `safeParse()`**，不抛异常，逻辑更清晰。

### TypeScript 类型推断

Zod 的杀手级功能——从 schema 自动推断 TypeScript 类型：

```ts
import { z } from 'zod';

const userSchema = z.object({
  id: z.number(),
  username: z.string(),
  email: z.string().email(),
});

type User = z.infer<typeof userSchema>;
// { id: number; username: string; email: string }
```

### Express 中间件封装

```js
function validate(schema) {
  return (req, res, next) => {
    const result = schema.safeParse(req.body);

    if (!result.success) {
      const errors = result.error.errors.map((e) => ({
        field: e.path.join('.'),
        message: e.message,
      }));
      return res.status(400).json({ error: '请求参数验证失败', details: errors });
    }

    req.body = result.data;
    next();
  };
}

// 也可验证 query 和 params
function validateQuery(schema) {
  return (req, res, next) => {
    const result = schema.safeParse(req.query);
    if (!result.success) {
      return res.status(400).json({ error: '查询参数验证失败', details: result.error.errors });
    }
    req.query = result.data;
    next();
  };
}

// 使用
app.get('/api/users', validateQuery(z.object({
  page: z.coerce.number().int().min(1).default(1),
  limit: z.coerce.number().int().min(1).max(100).default(10),
})), userController.list);
```

## 数据清洗 (Sanitization)

Joi 和 Zod 的验证会自动清洗数据（去除未定义字段、应用默认值），额外的清洗可以在验证后进行：

```js
// 清洗中间件 — 在验证之后执行
function sanitize(fields) {
  return (req, res, next) => {
    for (const field of fields) {
      if (typeof req.body[field] === 'string') {
        req.body[field] = req.body[field].trim();         // 去除首尾空格
        req.body[field] = req.body[field].replace(/\s+/g, ' '); // 合并多余空格
      }
    }
    next();
  };
}

// HTML 转义
function escapeHtml(str) {
  const map = { '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#x27;' };
  return str.replace(/[&<>"']/g, (c) => map[c]);
}
```

## Joi vs Zod 对比

| 维度       | Joi                      | Zod                         |
|-----------|--------------------------|-----------------------------|
| TypeScript | 需额外安装 @types/joi     | 原生支持，类型推断强大        |
| 学习曲线   | 稍陡，API 丰富            | 平缓，API 直观               |
| 包大小     | 较大（~500KB）            | 较小（~300KB）               |
| 自定义消息  | .messages() 灵活         | 直接传字符串                 |
| 生态       | Hapi.js 生态一哥          | tRPC / Next.js 标配          |
| 验证细节    | stripUnknown, abortEarly | 需手动 strip                |

## 实用示例：多层验证

```js
// schemas/user.js
const { z } = require('zod');

const createSchema = z.object({
  username: z.string().min(3).max(30),
  email: z.string().email(),
  password: z.string().min(8),
});

const updateSchema = z.object({
  username: z.string().min(3).max(30).optional(),
  email: z.string().email().optional(),
  bio: z.string().max(500).optional(),
});

const querySchema = z.object({
  page: z.coerce.number().int().min(1).default(1),
  limit: z.coerce.number().int().min(1).max(100).default(10),
  sort: z.enum(['newest', 'oldest', 'name']).default('newest'),
  search: z.string().max(100).optional(),
});

module.exports = { createSchema, updateSchema, querySchema };
```

## 总结

输入验证是 API 开发的必需品，不是可选项。Joi 适合复杂验证规则和传统 Express 项目，Zod 是 TypeScript 项目的最佳选择。核心模式：定义 Schema -> safeParse/validate -> 中间件封装 -> 使用清洗后的数据。验证要同时覆盖 body、query、params。
