# 定时任务调度 (node-cron)

许多业务场景需要定时执行自动化任务。node-cron 是 Node.js 中最流行的定时调度库，底层基于 cron 表达式。

## 安装

```bash
npm install node-cron
```

## Cron 表达式语法

```
 # ┌──────── second (0-59)        可选
 # │ ┌────── minute (0-59)
 # │ │ ┌──── hour (0-23)
 # │ │ │ ┌── day of month (1-31)
 # │ │ │ │ ┌ month (1-12 或简写)
 # │ │ │ │ │ ┌ day of week (0-7，0 和 7 都是周日)
 # │ │ │ │ │ │
 # * * * * * *
```

常用表达式：

| 表达式              | 含义                         |
|--------------------|------------------------------|
| `* * * * *`        | 每分钟执行                     |
| `0 * * * *`        | 每整点执行                     |
| `0 0 * * *`        | 每天凌晨执行                   |
| `0 0 * * 0`        | 每周日凌晨执行                 |
| `0 0 1 * *`        | 每月 1 号凌晨执行              |
| `0 8 * * 1-5`      | 工作日早上 8 点               |
| `*/30 * * * *`     | 每 30 分钟执行                |
| `0 0 * * *`        | 每天 00:00 执行               |

## 基本使用

```js
const cron = require('node-cron');

// 每天凌晨 2 点执行缓存清理
cron.schedule('0 2 * * *', () => {
  console.log('开始清理过期缓存...');
  // 实际清理逻辑
  cache.clearExpired();
  console.log('缓存清理完成');
});

console.log('定时任务已注册');
```

### 带秒的表达式

```js
// 第二个参数 { scheduled: false } 表示先不启动
const task = cron.schedule('0 30 10 * * 1-5', () => {
  console.log('工作日上午 10:30 执行');
}, { scheduled: false });

task.start(); // 手动启动
```

## 常见定时任务场景

### 1. 日报数据汇总

```js
// 每天凌晨 1 点生成昨日的业务报表
cron.schedule('0 1 * * *', async () => {
  const yesterday = new Date();
  yesterday.setDate(yesterday.getDate() - 1);
  const dateStr = yesterday.toISOString().split('T')[0];

  try {
    const stats = await reportService.generateDailyReport(dateStr);
    logger.info('日报生成完成', { date: dateStr, rows: stats.totalRows });
  } catch (err) {
    logger.error('日报生成失败', { date: dateStr, error: err.message });
  }
});
```

### 2. 过期数据清理

```js
// 每小时清理一次过期数据
cron.schedule('0 * * * *', async () => {
  try {
    const deleted = await db.query(
      'DELETE FROM sessions WHERE expires_at < NOW()',
    );
    logger.info('过期会话已清理', { count: deleted.rowCount });
  } catch (err) {
    logger.error('会话清理失败', { error: err.message });
  }
});
```

### 3. 定期数据同步

```js
// 每 30 分钟从外部 API 同步数据
cron.schedule('*/30 * * * *', async () => {
  try {
    const externalData = await fetch('https://api.external-service.com/data', {
      headers: { Authorization: `Bearer ${process.env.API_KEY}` },
    }).then((r) => r.json());

    await syncService.mergeData(externalData);
    logger.info('数据同步完成', { records: externalData.length });
  } catch (err) {
    logger.error('数据同步失败', { error: err.message });
  }
});
```

### 4. 健康检查上报

```js
// 每 5 分钟上报服务健康状态
cron.schedule('*/5 * * * *', async () => {
  const metrics = {
    memory: process.memoryUsage().heapUsed,
    uptime: process.uptime(),
    activeConnections: connectionPool.totalCount,
    timestamp: new Date().toISOString(),
  };

  try {
    await fetch('https://monitor.example.com/api/heartbeat', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(metrics),
    });
  } catch (err) {
    // 上报失败不影响主服务
    logger.warn('健康检查上报失败', { error: err.message });
  }
});
```

### 5. 临时文件清理

```js
const fs = require('fs');
const path = require('path');

const UPLOAD_TEMP_DIR = path.join(__dirname, '..', 'uploads', 'temp');
const TEMP_MAX_AGE = 24 * 60 * 60 * 1000; // 24 小时

// 每小时清理临时文件
cron.schedule('0 * * * *', () => {
  if (!fs.existsSync(UPLOAD_TEMP_DIR)) return;

  const now = Date.now();
  const files = fs.readdirSync(UPLOAD_TEMP_DIR);

  let deletedCount = 0;
  for (const file of files) {
    const filePath = path.join(UPLOAD_TEMP_DIR, file);
    const stats = fs.statSync(filePath);

    if (now - stats.mtimeMs > TEMP_MAX_AGE) {
      fs.unlinkSync(filePath);
      deletedCount++;
    }
  }

  if (deletedCount > 0) {
    logger.info('临时文件清理完成', { deleted: deletedCount });
  }
});
```

## 时区处理

```js
// 默认使用服务器时区
cron.schedule('0 0 * * *', task);

// 指定时区（如北京时间）
cron.schedule('0 0 * * *', task, {
  timezone: 'Asia/Shanghai',
});

// 多时区任务
cron.schedule('0 9 * * 1', sendAsiaReport, {
  timezone: 'Asia/Shanghai',
});

cron.schedule('0 9 * * 1', sendEuropeReport, {
  timezone: 'Europe/London',
});
```

## 管理定时任务

```js
// 创建任务
const task = cron.schedule('* * * * *', () => {
  console.log('每分钟执行');
}, { scheduled: false });

// 启动
task.start();
console.log('状态:', task.options.scheduled); // true

// 停止
task.stop();

// 验证任务是否在运行
if (!task.options.scheduled) {
  // 重新启动
  task.start();
}

// 运行时修改时间规则 — 创建新任务替换
function reschedule(task, newExpression) {
  task.stop();
  return cron.schedule(newExpression, task.callback);
}
```

## 错误处理

定时任务中的未捕获异常会导致任务停止，务必做好错误处理：

```js
// ❌ 错误 — 异步异常未捕获，可能导致任务静默失败
cron.schedule('0 * * * *', async () => {
  const data = await fetchData(); // 如果抛异常，任务崩溃
  await processData(data);
});

// ✅ 正确 — 完整的 try/catch
cron.schedule('0 * * * *', async () => {
  try {
    const data = await fetchData();
    await processData(data);
    logger.info('定时任务执行成功', { task: 'dataSync' });
  } catch (err) {
    logger.error('定时任务执行失败', {
      task: 'dataSync',
      error: err.message,
      stack: err.stack,
    });
    // 可选的告警通知
    // await sendAlert({ level: 'error', message: '数据同步失败' });
  }
});
```

## 完整的任务调度模块

```js
// tasks/scheduler.js
const cron = require('node-cron');
const logger = require('../lib/logger');

const tasks = [
  {
    name: 'dailyReport',
    schedule: '0 1 * * *',
    timezone: 'Asia/Shanghai',
    handler: require('./dailyReport'),
  },
  {
    name: 'cleanExpiredSessions',
    schedule: '0 * * * *',
    handler: require('./cleanSessions'),
  },
  {
    name: 'syncExternalData',
    schedule: '*/15 * * * *',
    handler: require('./syncData'),
  },
];

function startAll() {
  for (const task of tasks) {
    const job = cron.schedule(task.schedule, async () => {
      logger.info(`定时任务开始: ${task.name}`);
      try {
        await task.handler();
        logger.info(`定时任务完成: ${task.name}`);
      } catch (err) {
        logger.error(`定时任务失败: ${task.name}`, { error: err.message });
      }
    }, { timezone: task.timezone });

    logger.info(`任务已注册: ${task.name} (${task.schedule})`);
  }
}

module.exports = { startAll };
```

在应用入口中启动：

```js
// index.js
const scheduler = require('./tasks/scheduler');

scheduler.startAll();

app.listen(3000, () => {
  console.log('Server running on port 3000');
});
```

## node-schedule 替代方案

对于一次性延时任务，node-schedule 更合适：

```bash
npm install node-schedule
```

```js
const schedule = require('node-schedule');

// 5 分钟后执行一次
schedule.scheduleJob(Date.now() + 5 * 60 * 1000, () => {
  console.log('5 分钟到了');
});

// 指定日期时间执行
const date = new Date(2024, 11, 25, 10, 0, 0);
schedule.scheduleJob(date, () => {
  console.log('圣诞节当天 10:00 执行');
});
```

## 生产环境注意事项

1. **PM2 集群模式**：使用 PM2 的 `instances: max` 时，每个进程都会执行定时任务。需要加锁（Redis 分布式锁）确保只执行一次。
2. **避免长任务重叠**：如果上一次任务还没执行完，下一次触发时间就到了，任务会并发执行。在任务开始前加锁或检查运行状态。
3. **优雅关闭**：进程退出前停止所有定时任务。

```js
// 防止任务重叠
let isRunning = false;

cron.schedule('*/5 * * * *', async () => {
  if (isRunning) {
    logger.warn('上一次任务尚未完成，跳过本次执行');
    return;
  }
  isRunning = true;
  try {
    await doWork();
  } catch (err) {
    logger.error('执行失败', { error: err.message });
  } finally {
    isRunning = false;
  }
});
```

## 总结

node-cron 适合大多数定时任务场景，cron 表达式灵活强大。核心关注点：完整 try/catch 防止任务崩溃，防止任务重叠，集群环境加分布式锁，任务执行记录日志便于排查。
