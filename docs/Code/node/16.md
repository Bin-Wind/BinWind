# Node.js Stream 流处理详解

Stream（流）是 Node.js 最强大的抽象之一。它能以内存高效的方式处理数据 —— 将数据分成小块逐块处理，而不是一次性加载到内存。这对于处理大文件、网络传输和实时数据转换至关重要。

## Stream 的四种类型

| 类型 | 说明 | 典型例子 |
|------|------|----------|
| Readable | 可读流，数据来源 | `fs.createReadStream()`, HTTP request body |
| Writable | 可写流，数据去向 | `fs.createWriteStream()`, HTTP response |
| Duplex | 双工流，可读可写 | `net.Socket`, TLS socket |
| Transform | 转换流，读写中间做转换 | `zlib.createGzip()`, crypto streams |

## 文件复制：流 vs 缓冲

```javascript
// 方式一：全量读入内存（大文件会 OOM）
const fs = require('fs');
const data = fs.readFileSync('large.mp4');
fs.writeFileSync('copy.mp4', data);

// 方式二：使用 Stream（内存友好）
const fs = require('fs');
const readStream = fs.createReadStream('large.mp4');
const writeStream = fs.createWriteStream('copy.mp4');

readStream.pipe(writeStream);

writeStream.on('finish', () => {
  console.log('文件复制完成');
});
```

## fs.createReadStream() 与 fs.createWriteStream()

```javascript
const fs = require('fs');

const reader = fs.createReadStream('input.txt', {
  encoding: 'utf8',
  highWaterMark: 64 * 1024, // 64KB 内部缓冲区（默认值）
});

const writer = fs.createWriteStream('output.txt', { encoding: 'utf8' });

reader.on('data', (chunk) => {
  console.log(`收到 ${chunk.length} 字节`);
  writer.write(chunk);
});

reader.on('end', () => {
  writer.end();
  console.log('读取完成');
});

reader.on('error', (err) => {
  console.error('读取失败:', err.message);
});
```

## Pipe 管道与链式转换

`pipe()` 是流的核心方法，将可读流连接到可写流。

```javascript
const fs = require('fs');
const zlib = require('zlib');
const crypto = require('crypto');

// 链式管道：读取 → 压缩 → 加密 → 写入文件
fs.createReadStream('data.json')
  .pipe(zlib.createGzip())
  .pipe(crypto.createCipheriv('aes-256-cbc', key, iv))
  .pipe(fs.createWriteStream('data.enc.gz'))
  .on('finish', () => console.log('压缩加密完成'));
```

## 背压 (Backpressure)

当可写流处理速度跟不上可读流时，Node.js 会自动暂停可读流，防止内存爆炸。

```javascript
const fs = require('fs');

const reader = fs.createReadStream('big-file.log');
const writer = fs.createWriteStream('copy.log');

reader.on('data', (chunk) => {
  const canContinue = writer.write(chunk);
  if (!canContinue) {
    // 写入缓冲区已满，暂停读取
    reader.pause();
    writer.once('drain', () => {
      reader.resume(); // 缓冲区清空后恢复读取
    });
  }
});
```

手动处理背压时，检查 `write()` 的返回值并监听 `drain` 事件是关键。使用 `pipe()` 时这些由 Node.js 自动处理。

## 自定义 Transform 流

```javascript
const { Transform } = require('stream');

class UpperCaseTransform extends Transform {
  _transform(chunk, encoding, callback) {
    // chunk 默认是 Buffer
    const upper = chunk.toString().toUpperCase();
    this.push(upper);
    callback();
  }

  _flush(callback) {
    // 流结束时调用，可推送剩余数据
    console.log('转换完成');
    callback();
  }
}

// 使用自定义 Transform
fs.createReadStream('input.txt')
  .pipe(new UpperCaseTransform())
  .pipe(fs.createWriteStream('output.txt'));
```

## Readable.from() 与其他流工具

```javascript
const { Readable } = require('stream');

// 从可迭代对象创建可读流
const readable = Readable.from(['Apple', 'Banana', 'Cherry']);
readable.on('data', (chunk) => console.log(chunk.toString()));
// 输出: Apple, Banana, Cherry（每个元素作为一个 chunk）
```

## pipeline() 工具函数

`pipeline()` 比 `pipe()` 更安全，它会自动销毁未完成的流并传递错误。

```javascript
const { pipeline } = require('stream');
const fs = require('fs');
const zlib = require('zlib');

pipeline(
  fs.createReadStream('archive.tar'),
  zlib.createGzip(),
  fs.createWriteStream('archive.tar.gz'),
  (err) => {
    if (err) {
      console.error('管道处理失败:', err);
    } else {
      console.log('管道处理成功');
    }
  }
);

// Promise 版本（Node 15+）
const { pipeline } = require('stream/promises');

async function compress() {
  await pipeline(
    fs.createReadStream('data.txt'),
    zlib.createGzip(),
    fs.createWriteStream('data.txt.gz')
  );
  console.log('压缩完成');
}
```

## 实战：大型 CSV 转 JSON

```javascript
const fs = require('fs');
const { Transform, pipeline } = require('stream');
const { promisify } = require('util');
const pipelineAsync = promisify(pipeline);

class CsvToJsonTransform extends Transform {
  constructor() {
    super({ objectMode: true });
    this.headers = null;
    this.buffer = '';
  }

  _transform(chunk, encoding, callback) {
    this.buffer += chunk.toString();
    const lines = this.buffer.split('\n');
    this.buffer = lines.pop(); // 保留不完整的行

    if (!this.headers) {
      this.headers = lines[0].split(',');
      lines.shift();
    }

    for (const line of lines) {
      if (line.trim()) {
        const values = line.split(',');
        const obj = {};
        this.headers.forEach((h, i) => (obj[h.trim()] = values[i]?.trim()));
        this.push(JSON.stringify(obj) + '\n');
      }
    }
    callback();
  }

  _flush(callback) {
    if (this.buffer.trim()) {
      const values = this.buffer.split(',');
      const obj = {};
      this.headers.forEach((h, i) => (obj[h.trim()] = values[i]?.trim()));
      this.push(JSON.stringify(obj) + '\n');
    }
    callback();
  }
}

// 处理数 GB 的 CSV 文件
async function convertCsvToJson(inputPath, outputPath) {
  await pipelineAsync(
    fs.createReadStream(inputPath),
    new CsvToJsonTransform(),
    fs.createWriteStream(outputPath)
  );
  console.log('CSV → JSON 转换完成');
}

convertCsvToJson('data.csv', 'data.json');
```

## 总结

Stream 是处理大数据的利器：用很小的内存就能处理 GB 级别的文件。掌握四种流类型、`pipe()`/`pipeline()` 以及自定义 Transform，能让你写出高效且可靠的 Node.js 应用。始终优先使用 `pipeline()` 而非 `pipe()`，以获得更好的错误处理。
