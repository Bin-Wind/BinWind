# 单元测试 (Jest) 编写指南

Jest 是 Facebook 开源的 JavaScript 测试框架，也是 Node.js 项目中最流行的测试方案。它零配置、内置断言库和 Mock 功能，同时支持代码覆盖率统计。良好的单元测试能让你放心重构代码，是工程质量的基石。

## 安装与配置

```bash
npm install --save-dev jest
```

`package.json` 基础配置：

```json
{
  "scripts": {
    "test": "jest",
    "test:watch": "jest --watch",
    "test:coverage": "jest --coverage"
  },
  "jest": {
    "testEnvironment": "node",
    "testMatch": ["**/__tests__/**/*.test.js"],
    "collectCoverageFrom": ["src/**/*.js", "!src/**/*.test.js"]
  }
}
```

## 测试结构：describe() 与 test()

```javascript
// math.test.js
const { add, divide } = require('./math');

describe('add() 函数', () => {
  test('两正数相加', () => {
    expect(add(1, 2)).toBe(3);
  });

  test('负数相加', () => {
    expect(add(-1, -2)).toBe(-3);
  });
});

describe('divide() 函数', () => {
  test('正常除法', () => {
    expect(divide(10, 2)).toBe(5);
  });

  test('除数为零应抛出错误', () => {
    expect(() => divide(10, 0)).toThrow('除数不能为零');
  });
});
```

## 常用 Matchers（断言方法）

```javascript
// 精确匹配（基本类型）
expect(2 + 2).toBe(4);
expect('hello').toBe('hello');

// 深度相等（对象/数组）
expect({ a: 1, b: 2 }).toEqual({ a: 1, b: 2 });
expect([1, 2, 3]).toContain(2);

// 真值判断
expect(null).toBeNull();
expect(undefined).toBeUndefined();
expect(true).toBeTruthy();
expect(false).toBeFalsy();
expect(0).not.toBeTruthy();

// 数字比较
expect(0.1 + 0.2).toBeCloseTo(0.3);

// 字符串匹配
expect('Hello World').toMatch(/World/);

// 部分匹配对象
expect({ id: 1, name: 'Tom', age: 25 }).toMatchObject({ name: 'Tom' });

// 数组含特定元素
expect([{ id: 1 }, { id: 2 }]).toEqual(
  expect.arrayContaining([{ id: 2 }])
);
```

## 异步测试

```javascript
const { fetchUser } = require('./api');

describe('fetchUser', () => {
  // 方式一：async/await（推荐）
  test('获取用户数据', async () => {
    const user = await fetchUser(1);
    expect(user).toMatchObject({ id: 1, name: expect.any(String) });
  });

  // 方式二：resolves / rejects
  test('成功时解析用户', () => {
    return expect(fetchUser(1)).resolves.toHaveProperty('name');
  });

  test('失败时抛出错误', () => {
    return expect(fetchUser(999)).rejects.toThrow('用户不存在');
  });

  // 方式三：done 回调（仅用于回调风格的异步代码）
  test('回调风格', (done) => {
    function callback(err, data) {
      try {
        expect(data).toBeDefined();
        expect(err).toBeNull();
        done();
      } catch (error) {
        done(error);
      }
    }
    someCallbackApi(callback);
  });
});
```

## Mock 函数

```javascript
// jest.fn() — 创建模拟函数
test('mock 函数基本用法', () => {
  const fn = jest.fn();
  fn('hello');
  fn('world');

  expect(fn).toHaveBeenCalledTimes(2);
  expect(fn).toHaveBeenCalledWith('hello');
  expect(fn.mock.calls[0][0]).toBe('hello');
});

// 模拟返回值
test('mock 返回值', () => {
  const fn = jest.fn()
    .mockReturnValueOnce(10)
    .mockReturnValueOnce(20)
    .mockReturnValue(100);

  expect(fn()).toBe(10);
  expect(fn()).toBe(20);
  expect(fn()).toBe(100);
  expect(fn()).toBe(100);
});

// jest.spyOn() — 监听真实对象的方法
const db = require('./db');

test('spyOn 监听', () => {
  const spy = jest.spyOn(db, 'query').mockResolvedValue([{ id: 1 }]);

  // 执行业务逻辑...
  expect(spy).toHaveBeenCalledTimes(1);

  spy.mockRestore(); // 恢复原始实现
});
```

## Mock 模块与 HTTP 调用

```javascript
// 方式一：jest.mock() 自动模拟整个模块
jest.mock('axios');
const axios = require('axios');
const { getUserPosts } = require('./posts');

test('获取用户帖子', async () => {
  axios.get.mockResolvedValue({
    data: [{ id: 1, title: 'Hello' }],
  });

  const posts = await getUserPosts(1);
  expect(posts).toHaveLength(1);
  expect(axios.get).toHaveBeenCalledWith('/users/1/posts');
});

// 方式二：nock 库拦截 HTTP（不 mock 模块，更真实）
const nock = require('nock');

test('使用 nock 拦截 HTTP', async () => {
  nock('https://api.example.com')
    .get('/users/1/posts')
    .reply(200, [{ id: 1, title: 'Hello' }]);

  const posts = await getUserPosts(1);
  expect(posts).toHaveLength(1);
});
```

## 测试 Express 路由

```javascript
const request = require('supertest');
const express = require('express');

// 假设 app.js 导出 express 实例
const app = require('./app');

describe('GET /api/users', () => {
  test('返回用户列表', async () => {
    const res = await request(app)
      .get('/api/users')
      .expect('Content-Type', /json/)
      .expect(200);

    expect(res.body).toEqual({
      success: true,
      data: expect.any(Array),
    });
  });
});

describe('POST /api/users', () => {
  test('创建用户成功', async () => {
    const res = await request(app)
      .post('/api/users')
      .send({ name: 'Tom', email: 'tom@example.com' })
      .expect(201);

    expect(res.body.data).toMatchObject({ name: 'Tom' });
  });
});
```

## 代码覆盖率

```bash
npx jest --coverage
```

输出示例：

```
File       | % Stmts | % Branch | % Funcs | % Lines
-----------|---------|----------|---------|--------
All files  |   85.71 |       80 |      75 |   85.71
 src/      |       0 |        0 |       0 |       0
  app.js   |     100 |      100 |     100 |     100
  user.js  |   83.33 |       75 |   66.66 |   83.33
```

关键指标解读：

- **% Stmts** (语句覆盖率)：代码中每条语句执行的比例
- **% Branch** (分支覆盖率)：if / else / switch 等分支都被执行的比例
- **% Funcs** (函数覆盖率)：每个函数是否都被调用过
- **% Lines** (行覆盖率)：每行代码被执行的比例

追求 100% 覆盖率不现实也不必要，目标通常定在 80% 以上即可。

## 总结

Jest 让单元测试变得简单直接。记住核心模式：`describe` 分组 → `test` 用例 → `expect` 断言。善用 `jest.fn()` 隔离外部依赖、用 `supertest` 测试 HTTP 接口、用 `--coverage` 量化测试质量。写测试不仅是为了找 bug，更是为了让自己和团队成员敢于修改代码。
