# child_process 子进程管理

Node.js 是单线程的，但可以通过 `child_process` 模块创建子进程来执行系统命令、运行 Python 脚本或并行处理 CPU 密集型任务。该模块提供了四种方法：`exec`、`spawn`、`execFile` 和 `fork`，各有适用场景。

## 四种方法对比

| 方法 | 使用场景 | 输出方式 | 默认 Shell | 支持 IPC |
|------|----------|----------|------------|----------|
| `exec()` | 短命令，输出小 | Buffer 全量回调 | 是 | 否 |
| `spawn()` | 长任务，实时流 | Stream | 否 | 否 |
| `execFile()` | 直接运行可执行文件 | Buffer 全量回调 | 否 | 否 |
| `fork()` | Node.js 子进程通信 | - | 否 | 是 |

## exec() — 执行短命令

`exec()` 启动一个 shell 来执行命令，缓冲全部输出并在完成后回调。适合执行返回结果不大的命令。

```javascript
const { exec } = require('child_process');

exec('ls -la /tmp', { maxBuffer: 1024 * 1024 }, (error, stdout, stderr) => {
  if (error) {
    console.error('执行出错:', error.message);
    return;
  }
  if (stderr) {
    console.error('stderr:', stderr);
    return;
  }
  console.log('输出:', stdout);
});

// Promise 封装
const { promisify } = require('util');
const execAsync = promisify(exec);

async function listFiles(dir) {
  try {
    const { stdout, stderr } = await execAsync(`ls -la ${dir}`);
    return stdout;
  } catch (err) {
    console.error('错误:', err.message);
    throw err;
  }
}
```

## spawn() — 运行长时间进程

`spawn()` 不调用 shell，直接启动进程，通过 Stream 返回输出。适合处理大输出和长时间运行的任务。

```javascript
const { spawn } = require('child_process');

// 实时流式输出
const child = spawn('ffmpeg', [
  '-i', 'input.mp4',
  '-vf', 'scale=1280:720',
  '-c:a', 'copy',
  'output.mp4',
]);

child.stdout.on('data', (data) => {
  console.log(`stdout: ${data}`);
});

child.stderr.on('data', (data) => {
  console.error(`stderr: ${data}`);
});

child.on('close', (code) => {
  console.log(`进程退出，退出码: ${code}`);
});

// 管道连接多个进程（等价于: cat file.txt | grep keyword）
const cat = spawn('cat', ['data.txt']);
const grep = spawn('grep', ['ERROR']);

cat.stdout.pipe(grep.stdin);

grep.stdout.on('data', (data) => {
  console.log(`匹配结果: ${data}`);
});
```

## execFile() — 直接运行可执行文件

跳过 shell 直接执行可执行文件，更安全（不会被 shell 注入攻击），性能略好。

```javascript
const { execFile } = require('child_process');

// 运行 Python 脚本
execFile('python3', ['process_image.py', 'photo.jpg'], (error, stdout, stderr) => {
  if (error) {
    console.error('执行失败:', error);
    return;
  }
  console.log('脚本输出:', stdout);
});

// 运行 Node.js 脚本（也可以用，但 fork 更好）
execFile('node', ['--version'], (error, stdout) => {
  console.log('Node 版本:', stdout.trim());
});
```

## fork() — Node.js 专用子进程

`fork()` 是 `spawn()` 的特殊版本，专门用于创建 Node.js 子进程并建立 IPC 通信通道。

```javascript
// parent.js
const { fork } = require('child_process');

const child = fork('./worker.js');

// 向子进程发送消息
child.send({ task: 'compute', data: [1, 2, 3, 4, 5] });

// 接收子进程消息
child.on('message', (message) => {
  console.log('子进程结果:', message.result);
});

child.on('exit', (code) => {
  console.log(`子进程退出，代码: ${code}`);
});
```

```javascript
// worker.js
process.on('message', (msg) => {
  console.log(`收到任务: ${msg.task}`);

  let sum = 0;
  for (let i = 0; i < 1e9; i++) {
    sum += i;
  }

  // 发送结果回父进程
  process.send({ result: sum });
});
```

## 关键选项详解

```javascript
const child = spawn('node', ['app.js'], {
  cwd: '/path/to/project',     // 工作目录
  env: { ...process.env, NODE_ENV: 'production' }, // 环境变量
  timeout: 10000,                       // 超时自动 kill（ms）
  shell: '/bin/bash',                   // 指定 shell
  stdio: ['pipe', 'pipe', 'pipe'],      // stdin / stdout / stderr 配置
});
```

### stdio 继承

```javascript
// 让子进程共享父进程的 stdio（输出直接显示在终端）
spawn('npm', ['run', 'build'], { stdio: 'inherit' });

// 捕获 stderr 但忽略 stdout
spawn('node', ['script.js'], {
  stdio: ['pipe', 'ignore', 'pipe'],
});
```

## 错误处理完整示例

```javascript
const { spawn } = require('child_process');

function runCommand(cmd, args, options = {}) {
  return new Promise((resolve, reject) => {
    const child = spawn(cmd, args, options);

    let stdout = '';
    let stderr = '';

    child.stdout?.on('data', (data) => (stdout += data.toString()));
    child.stderr?.on('data', (data) => (stderr += data.toString()));

    child.on('error', (err) => {
      reject(new Error(`无法启动进程: ${err.message}`));
    });

    child.on('close', (code) => {
      if (code === 0) {
        resolve({ stdout, stderr });
      } else {
        reject(new Error(`进程退出码: ${code}\n${stderr}`));
      }
    });

    // 超时处理
    if (options.timeout) {
      setTimeout(() => {
        child.kill('SIGTERM');
        reject(new Error('进程超时'));
      }, options.timeout);
    }
  });
}

// 使用示例
runCommand('python3', ['train_model.py', '--epochs=50'], { timeout: 60000 })
  .then(({ stdout }) => console.log('训练结果:', stdout))
  .catch((err) => console.error('训练失败:', err.message));
```

## 实战：图片批量压缩

```javascript
const { fork } = require('child_process');
const path = require('path');
const fs = require('fs');

const images = fs.readdirSync('./images').filter(f => /\.(jpg|png)$/i.test(f));

// 启动多个 worker 并行处理（每个 CPU 一个）
const os = require('os');
const numWorkers = os.cpus().length;

let completed = 0;
const chunkSize = Math.ceil(images.length / numWorkers);

for (let i = 0; i < numWorkers; i++) {
  const chunk = images.slice(i * chunkSize, (i + 1) * chunkSize);
  if (chunk.length === 0) continue;

  const worker = fork('./compress-worker.js');
  worker.send({ images: chunk, quality: 80 });

  worker.on('message', (msg) => {
    completed += msg.processed;
    console.log(`进度: ${completed}/${images.length}`);
    if (completed >= images.length) {
      console.log('全部压缩完成');
    }
  });
}
```

## 总结

- `exec()` 适合一次性获取命令输出的场景（如 `git log`）
- `spawn()` 适合流式处理输出和长时间运行的进程（如视频转码）
- `execFile()` 直接执行文件，避免 shell 注入，更安全
- `fork()` 专用于 Node.js 子进程 IPC，实现 CPU 密集型任务的多进程并行

选择哪个方法取决于任务特征：输出大小、运行时长以及是否需要与子进程通信。
