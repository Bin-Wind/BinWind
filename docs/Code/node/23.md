# 安全最佳实践 (Helmet, Rate Limiting)

Web 应用的安全漏洞每年造成数十亿美元损失。对于 Node.js 后端来说，安全不是"加上就完事"的功能，而是贯穿开发全程的设计。本文聚焦于三个必须配置的防线：Helmet（安全头）、Rate Limiting（限流）和 CORS（跨域控制），外加实用的安全检查清单。

## Helmet — 一键设置安全响应头

Helmet 为 Express 应用设置多个 HTTP 安全头，屏蔽常见 Web 漏洞。

```bash
npm install helmet
```

```javascript
const express = require('express');
const helmet = require('helmet');

const app = express();

// 基础用法：使用默认配置
app.use(helmet());

// 高级用法：逐项开关
app.use(helmet({
  contentSecurityPolicy: {
    directives: {
      defaultSrc: ["'self'"],
      scriptSrc: ["'self'", 'cdn.example.com'],
      styleSrc: ["'self'", "'unsafe-inline'"],
      imgSrc: ["'self'", 'data:', 'images.example.com'],
    },
  },
  xssFilter: true,              // X-XSS-Protection
  frameguard: { action: 'deny' }, // X-Frame-Options
  hsts: {                       // Strict-Transport-Security
    maxAge: 63072000,           // 2 年
    includeSubDomains: true,
    preload: true,
  },
  referrerPolicy: { policy: 'strict-origin-when-cross-origin' },
}));
```

### Helmet 设置的安全头一览

| 响应头 | 作用 | 默认值 |
|--------|------|--------|
| Content-Security-Policy | 防止 XSS、数据注入 | `default-src 'self'` |
| X-Content-Type-Options | 禁止 MIME 类型嗅探 | `nosniff` |
| X-Frame-Options | 防止点击劫持 | `SAMEORIGIN` |
| Strict-Transport-Security | 强制 HTTPS | `max-age=15552000` |
| X-DNS-Prefetch-Control | 控制 DNS 预取 | `off` |
| X-Permitted-Cross-Domain-Policies | 限制 Adobe Flash/PDF | `none` |
| Referrer-Policy | 控制 Referer 信息 | `no-referrer` |
| X-XSS-Protection | XSS 过滤（旧浏览器） | `0` |

## Rate Limiting — 接口限流防滥用

`express-rate-limit` 是 Express 最流行的限流中间件。

```bash
npm install express-rate-limit
```

### 基础限流

```javascript
const rateLimit = require('express-rate-limit');

// 全局限流：15 分钟内最多 100 次请求
const globalLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,  // 15 分钟
  max: 100,
  standardHeaders: true,     // 返回 RateLimit-* 头
  legacyHeaders: false,      // 禁用 X-RateLimit-* 头
  message: { error: '请求过于频繁，请稍后再试' },
});

app.use('/api/', globalLimiter);
```

### 分级限流

```javascript
// 严格限制：登录接口（防暴力破解）
const loginLimiter = rateLimit({
  windowMs: 60 * 1000, // 1 分钟
  max: 5,              // 最多 5 次尝试
  skipSuccessfulRequests: true, // 成功后不计入
  message: { error: '登录尝试过于频繁，请1分钟后重试' },
});
app.use('/api/login', loginLimiter);

// 宽松限制：静态资源 API
const apiLimiter = rateLimit({
  windowMs: 60 * 1000, // 1 分钟
  max: 300,
});
app.use('/api/public', apiLimiter);
```

### 基于 Redis 的分布式限流（多进程场景）

```javascript
const RedisStore = require('rate-limit-redis');
const Redis = require('ioredis');

const redisClient = new Redis({ host: 'localhost', port: 6379 });

const distributedLimiter = rateLimit({
  store: new RedisStore({
    sendCommand: (...args) => redisClient.call(...args),
  }),
  windowMs: 15 * 60 * 1000,
  max: 100,
});
```

## CORS — 跨域资源安全

```bash
npm install cors
```

```javascript
const cors = require('cors');

// 基础用法：允许所有来源（开发环境）
app.use(cors());

// 生产环境：白名单
const allowedOrigins = [
  'https://myapp.com',
  'https://admin.myapp.com',
];

app.use(cors({
  origin: (origin, callback) => {
    // 允许同源请求（origin 为 undefined）
    if (!origin || allowedOrigins.includes(origin)) {
      callback(null, true);
    } else {
      callback(new Error('不允许的跨域来源'));
    }
  },
  methods: ['GET', 'POST', 'PUT', 'DELETE'],
  allowedHeaders: ['Content-Type', 'Authorization'],
  credentials: true,                // 允许携带 Cookie
  maxAge: 86400,                    // 预检请求缓存 24h
}));
```

## 输入验证（参考文章 11）

```javascript
// 永远不要信任用户输入
app.post('/api/users', (req, res) => {
  // ❌ 直接使用
  // const name = req.body.name;

  // ✅ 验证和净化
  const { name, email, age } = req.body;

  if (typeof name !== 'string' || name.length > 100) {
    return res.status(400).json({ error: '无效的用户名' });
  }

  if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) {
    return res.status(400).json({ error: '无效的邮箱' });
  }

  const parsedAge = parseInt(age, 10);
  if (isNaN(parsedAge) || parsedAge < 0 || parsedAge > 150) {
    return res.status(400).json({ error: '无效的年龄' });
  }

  // 安全地使用
  db.query('INSERT INTO users (name, email, age) VALUES (?, ?, ?)',
    [name, email, parsedAge]
  );
});
```

## 生产环境安全清单

```javascript
// 生产环境 Checklist
const isProduction = process.env.NODE_ENV === 'production';

// 1. 隐藏技术栈信息
app.disable('x-powered-by'); // 移除 X-Powered-By: Express

// 2. 全局错误处理：不暴露堆栈
app.use((err, req, res, next) => {
  console.error(err.stack);

  res.status(err.status || 500).json({
    error: isProduction
      ? '服务器内部错误'
      : err.message,
    // 生产环境不返回 stack
  });
});

// 3. 请求体大小限制
app.use(express.json({ limit: '1mb' }));
app.use(express.urlencoded({ extended: true, limit: '1mb' }));

// 4. 安全的 Cookie 配置
res.cookie('session', token, {
  httpOnly: true,       // 防止 JS 读取
  secure: isProduction, // 仅通过 HTTPS 发送
  sameSite: 'strict',   // 防止 CSRF
  maxAge: 3600000,
});
```

## 完整安全配置模板

```javascript
const express = require('express');
const helmet = require('helmet');
const cors = require('cors');
const rateLimit = require('express-rate-limit');

const app = express();

// 1. 安全头
app.use(helmet());

// 2. CORS 白名单
app.use(cors({
  origin: process.env.ALLOWED_ORIGINS?.split(',') || 'http://localhost:5173',
  credentials: true,
}));

// 3. 限流
app.use('/api/', rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 100,
}));

// 4. 隐藏框架信息
app.disable('x-powered-by');

// 5. 请求体大小限制
app.use(express.json({ limit: '1mb' }));

// 6. 全局安全错误处理
app.use((err, req, res, next) => {
  console.error(err);
  res.status(500).json({ error: 'Internal Server Error' });
});

app.listen(3000);
```

## 总结

安全不是一劳永逸的，而是持续对抗。三板斧必装：**Helmet** 堵住常见漏洞的门、**Rate Limiting** 防滥用、**CORS** 管好跨域访问。再加上输入验证、依赖审计和环境机密管理，你的 Node.js 应用就有了基本的安全防线。定期运行 `npm audit`、关注新的 CVE，安全意识比任何工具都更重要。
