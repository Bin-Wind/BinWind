# Express.js 搭建 RESTful API 服务

Express 是 Node.js 最流行的 Web 框架。本文从零搭建一个完整的 RESTful API 服务。

## 项目初始化

```bash
mkdir express-api && cd express-api
npm init -y
npm install express
```

## 最小可用服务器

```js
// index.js
const express = require('express');
const app = express();
const PORT = process.env.PORT || 3000;

app.get('/', (req, res) => {
  res.json({ message: 'Hello, Express!' });
});

app.listen(PORT, () => {
  console.log(`Server running on http://localhost:${PORT}`);
});
```

```bash
node index.js
# Server running on http://localhost:3000
```

## JSON Body 解析

```js
// 解析 Content-Type: application/json 的请求体
app.use(express.json());
```

## 路由方法

```js
// GET — 获取资源
app.get('/api/users', (req, res) => {
  res.json(users);
});

// POST — 创建资源
app.post('/api/users', (req, res) => {
  const user = { id: Date.now(), ...req.body };
  users.push(user);
  res.status(201).json(user);
});

// PUT — 全量更新
app.put('/api/users/:id', (req, res) => {
  const index = users.findIndex((u) => u.id === req.params.id);
  if (index === -1) return res.status(404).json({ error: 'Not found' });
  users[index] = { id: req.params.id, ...req.body };
  res.json(users[index]);
});

// DELETE — 删除资源
app.delete('/api/users/:id', (req, res) => {
  const index = users.findIndex((u) => u.id === req.params.id);
  if (index === -1) return res.status(404).json({ error: 'Not found' });
  const [deleted] = users.splice(index, 1);
  res.json(deleted);
});
```

### 请求参数获取

```js
app.get('/api/users/:id/posts/:postId', (req, res) => {
  // 路由参数
  console.log(req.params);  // { id: '1', postId: '2' }

  // 查询字符串 ?page=1&limit=10
  console.log(req.query);   // { page: '1', limit: '10' }

  // 请求体（需 express.json() 中间件）
  console.log(req.body);    // { name: 'John', email: 'john@example.com' }
});
```

### 路由层 (routes/users.js)

```js
const express = require('express');
const router = express.Router();
const usersController = require('../controllers/users');

router.get('/', usersController.list);
router.post('/', usersController.create);
router.get('/:id', usersController.getById);
router.put('/:id', usersController.update);
router.delete('/:id', usersController.remove);

module.exports = router;
```

### 控制器层 (controllers/users.js)

```js
const User = require('../models/user');

exports.list = (req, res) => {
  const { page = 1, limit = 10 } = req.query;
  const data = User.findAll({ page: +page, limit: +limit });
  res.json({ total: data.length, data });
};

exports.create = (req, res) => {
  const { name, email } = req.body;
  if (!name || !email) {
    return res.status(400).json({ error: 'name 和 email 为必填字段' });
  }
  const user = User.create({ name, email });
  res.status(201).json(user);
};

exports.getById = (req, res) => {
  const user = User.findById(req.params.id);
  if (!user) return res.status(404).json({ error: '用户不存在' });
  res.json(user);
};

exports.update = (req, res) => {
  const user = User.update(req.params.id, req.body);
  if (!user) return res.status(404).json({ error: '用户不存在' });
  res.json(user);
};

exports.remove = (req, res) => {
  const deleted = User.delete(req.params.id);
  if (!deleted) return res.status(404).json({ error: '用户不存在' });
  res.status(204).send();
};
```

### 模型层 — 内存数据 (models/user.js)

```js
const users = [];

class User {
  static findAll({ page, limit }) {
    const start = (page - 1) * limit;
    return users.slice(start, start + limit);
  }

  static findById(id) {
    return users.find((u) => u.id === id);
  }

  static create(data) {
    const user = { id: String(Date.now()), ...data, createdAt: new Date() };
    users.push(user);
    return user;
  }

  static update(id, data) {
    const index = users.findIndex((u) => u.id === id);
    if (index === -1) return null;
    users[index] = { ...users[index], ...data, updatedAt: new Date() };
    return users[index];
  }

  static delete(id) {
    const index = users.findIndex((u) => u.id === id);
    if (index === -1) return null;
    return users.splice(index, 1)[0];
  }
}

module.exports = User;
```

### 入口文件整合 (index.js)

```js
const express = require('express');
const usersRouter = require('./routes/users');
const errorHandler = require('./middleware/errorHandler');

const app = express();

app.use(express.json());
app.use('/api/users', usersRouter);

// 全局错误处理必须放在路由之后
app.use(errorHandler);

const PORT = 3000;
app.listen(PORT, () => console.log(`Server running on port ${PORT}`));
```

## 错误处理中间件 (middleware/errorHandler.js)

```js
module.exports = (err, req, res, next) => {
  console.error(err.stack);
  res.status(err.status || 500).json({
    error: err.message || '服务器内部错误',
  });
};
```

```bash
# 更新用户

```

## 总结

核心要点：`express.json()` 解析请求体，路由分离到独立文件，控制器处理业务逻辑，模型层抽象数据操作。错误处理中间件统一兜底。这个结构可作为中型项目的起点。
