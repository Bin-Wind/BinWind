# 中间件机制深入理解

中间件是 Express 的核心概念。理解中间件的执行顺序和分类是掌握 Express 的关键。

## 什么是中间件

中间件是接收 `(req, res, next)` 的函数，在请求-响应周期中按顺序执行。

```js
app.use((req, res, next) => {
  console.log(`${req.method} ${req.url}`);
  next(); // 传递给下一个中间件或路由处理函数
});
```

调用 `next()` 将控制权交给下一个中间件。不调用 `next()` 则请求挂起。调用 `next(err)` 则跳过所有常规中间件，进入错误处理流程。

## 执行顺序

中间件按**注册顺序**执行。请求到达后，从第一个匹配的中间件开始，逐个调用。

```js
app.use((req, res, next) => {
  console.log('1. 记录请求开始');
  next();
  console.log('8. 响应已发送，但这段仍然执行');
});

app.get('/api/data', (req, res, next) => {
  console.log('2. 进入路由');
  next();
});

app.use((req, res, next) => {
  console.log('3. 路由后的中间件');
  res.json({ message: 'ok' }); // 发送响应，请求结束
});

// 这段不会执行（响应已发送）
app.use((req, res, next) => {
  console.log('这段不会打印');
});
```

输出：
```
1. 记录请求开始
2. 进入路由
3. 路由后的中间件
8. 响应已发送，但这段仍然执行
```

> 注意：`next()` 之后的代码仍然会执行，只是响应已经发送。

## 中间件分类

### 应用级中间件 (Application-level)

通过 `app.use()` 或 `app.METHOD()` 绑定到 Express 实例，可指定路径前缀。

```js
app.use('/api', (req, res, next) => {
  console.log('只匹配 /api 开头的请求');
  next();
});

app.get('/api/users', (req, res) => {
  res.json([]);
});
```

### 路由级中间件 (Router-level)

用法和应用级一致，但绑定在 `express.Router()` 实例上。

```js
const router = express.Router();

router.use((req, res, next) => {
  console.log('Router 级别的中间件');
  next();
});

router.get('/profile', (req, res) => {
  res.json({ name: 'John' });
});

app.use('/user', router);
```

### 内置中间件

```js
// 解析 JSON 请求体
app.use(express.json());

// 解析 URL 编码的请求体（表单提交）
app.use(express.urlencoded({ extended: true }));

// 托管静态文件
app.use(express.static('public'));

// 多目录静态文件
app.use(express.static('uploads'));
app.use(express.static('assets'));
```

### 错误处理中间件

错误处理中间件有四个参数，Express 通过参数数量识别。

```js
app.use((err, req, res, next) => {
  console.error(err.stack);

  const status = err.status || 500;
  const message = status === 500 ? '服务器内部错误' : err.message;

  res.status(status).json({ error: message });
});
```

错误传递方式：

```js
app.get('/api/data', (req, res, next) => {
  try {
    // 可能出错的代码
    const result = riskyOperation();
    res.json(result);
  } catch (err) {
    next(err); // 传递给错误处理中间件
  }
});

// 或者直接抛出（Express 会自动捕获同步异常）
app.get('/api/sync', (req, res) => {
  throw new Error('同步错误会被 Express 自动捕获');
});
```

## 自定义中间件实战

### 日志中间件

```js
function logger(req, res, next) {
  const start = Date.now();
  res.on('finish', () => {
    const duration = Date.now() - start;
    console.log(`${req.method} ${req.originalUrl} ${res.statusCode} ${duration}ms`);
  });
  next();
}

app.use(logger);
```

### 认证中间件

```js
function auth(req, res, next) {
  const token = req.headers.authorization?.replace('Bearer ', '');
  if (!token) {
    return res.status(401).json({ error: '未提供认证令牌' });
  }
  try {
    req.user = jwt.verify(token, process.env.JWT_SECRET);
    next();
  } catch (err) {
    return res.status(401).json({ error: '令牌无效或已过期' });
  }
}

// 只对特定路由生效
app.use('/api/admin', auth);
app.use('/api/admin/users', adminRouter);
```

### 请求验证中间件

```js
function validate(schema) {
  return (req, res, next) => {
    const { error, value } = schema.validate(req.body);
    if (error) {
      return res.status(400).json({ error: error.details[0].message });
    }
    req.body = value; // 使用清洗后的数据
    next();
  };
}

const createUserSchema = Joi.object({
  name: Joi.string().min(2).max(50).required(),
  email: Joi.string().email().required(),
  age: Joi.number().min(0).max(150),
});

app.post('/api/users', validate(createUserSchema), (req, res) => {
  // req.body 已经过验证和清洗
  res.json(req.body);
});
```

### 限流中间件（简单实现）

```js
const requests = new Map();

function rateLimiter(maxRequests = 100, windowMs = 60000) {
  return (req, res, next) => {
    const ip = req.ip;
    const now = Date.now();

    if (!requests.has(ip)) {
      requests.set(ip, []);
    }
    const timestamps = requests.get(ip).filter((t) => now - t < windowMs);
    timestamps.push(now);
    requests.set(ip, timestamps);

    if (timestamps.length > maxRequests) {
      return res.status(429).json({ error: '请求过于频繁，请稍后重试' });
    }
    next();
  };
}

app.use('/api', rateLimiter(100, 60000));
```

const cors = require('cors');
const helmet = require('helmet');

app.use(cors({

app.use(helmet());

app.use(morgan('combined'));

app.use(compression());

## 完整的中间件装配顺序

```js
const express = require('express');
const app = express();

// 1. 安全头
app.use(helmet());

// 2. CORS
app.use(cors());

// 3. 请求日志
app.use(morgan('dev'));

// 4. 压缩
app.use(compression());

// 5. Body 解析
app.use(express.json());
app.use(express.urlencoded({ extended: true }));

// 6. 自定义日志
app.use(logger);

// 7. 路由
app.use('/api', apiRouter);
app.use('/api/admin', auth, adminRouter);

// 8. 静态文件
app.use(express.static('public'));

// 9. 404 处理
app.use((req, res) => {
  res.status(404).json({ error: '请求的资源不存在' });
});

// 10. 全局错误处理（必须最后）
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(err.status || 500).json({ error: '服务器错误' });
});
```

## 总结

中间件的核心是 `(req, res, next)` 链式调用。理解执行顺序和 next() 的语义是关键：调用 next() 继续、不调用则暂停、传入 Error 则跳到错误处理。合理的中间件装配顺序：安全 -> 日志 -> 解析 -> 业务路由 -> 静态 -> 404 -> 错误处理。
