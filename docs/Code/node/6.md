# 文件上传处理 (Multer)

Multer 是 Express 生态中处理 multipart/form-data 文件上传的标准中间件。本文覆盖单文件、多文件上传以及常见业务场景。

## 安装

```bash
npm install multer
```

## 基础单文件上传

```js
const express = require('express');
const multer = require('multer');
const path = require('path');

const app = express();

const upload = multer({ dest: 'uploads/' });

app.post('/upload', upload.single('avatar'), (req, res) => {
  console.log(req.file); // 上传的文件信息
  console.log(req.body); // 其他表单字段
  res.json({
    message: '上传成功',
    file: req.file,
  });
});
```

`req.file` 包含的关键字段：

```js
{
  fieldname: 'avatar',       // 表单字段名
  originalname: 'photo.jpg', // 客户端原始文件名
  encoding: '7bit',
  mimetype: 'image/jpeg',
  destination: 'uploads/',   // 存储目录
  filename: 'abc123.jpg',   // 存储的文件名（随机生成）
  path: 'uploads/abc123.jpg',
  size: 102400              // 字节数
}
```

## 磁盘存储配置

默认 `dest` 选项使用随机文件名。生产环境通常需要自定义存储策略：

```js
const storage = multer.diskStorage({
  destination(req, file, cb) {
    cb(null, 'uploads/');
  },
  filename(req, file, cb) {
    // 保留原始扩展名，使用时间戳+随机数防冲突
    const ext = path.extname(file.originalname);
    const uniqueName = `${Date.now()}-${Math.round(Math.random() * 1e9)}${ext}`;
    cb(null, uniqueName);
  },
});

const upload = multer({ storage });
```

## 文件过滤

限制上传文件类型和大小：

```js
const upload = multer({
  storage,
  limits: {
    fileSize: 5 * 1024 * 1024, // 5MB
    files: 10,                  // 最多 10 个文件
  },
  fileFilter(req, file, cb) {
    // 只允许图片
    const allowedMimes = ['image/jpeg', 'image/png', 'image/gif', 'image/webp'];
    if (allowedMimes.includes(file.mimetype)) {
      cb(null, true);
    } else {
      cb(new Error('只允许上传 JPG / PNG / GIF / WebP 格式的图片'), false);
    }
  },
});
```

## 多文件上传

### 同名字段多文件 — upload.array()

```html
<form enctype="multipart/form-data">
  <input type="file" name="photos" multiple />
</form>
```

```js
app.post('/upload/photos', upload.array('photos', 5), (req, res) => {
  // req.files 是文件数组
  const urls = req.files.map((f) => `/uploads/${f.filename}`);
  res.json({ message: `${req.files.length} 个文件上传成功`, urls });
});
```

### 不同字段多文件 — upload.fields()

```html
<form>
  <input type="file" name="avatar" />
  <input type="file" name="idCard" />
</form>
```

```js
app.post('/upload/profile', upload.fields([
  { name: 'avatar', maxCount: 1 },
  { name: 'idCard', maxCount: 1 },
]), (req, res) => {
  const avatar = req.files['avatar']?.[0];
  const idCard = req.files['idCard']?.[0];
  res.json({ avatar, idCard });
});
```

## Multer 错误处理

```js
const multer = require('multer');

app.post('/upload', (req, res, next) => {
  upload.single('file')(req, res, (err) => {
    if (err instanceof multer.MulterError) {
      // Multer 自身错误
      switch (err.code) {
        case 'LIMIT_FILE_SIZE':
          return res.status(400).json({ error: '文件大小超过限制（最大 5MB）' });
        case 'LIMIT_FILE_COUNT':
          return res.status(400).json({ error: '文件数量超过限制' });
        case 'LIMIT_UNEXPECTED_FILE':
          return res.status(400).json({ error: '意外的文件字段' });
        default:
          return res.status(400).json({ error: err.message });
      }
    } else if (err) {
      // 自定义错误（如 fileFilter 抛出的）
      return res.status(400).json({ error: err.message });
    }
    next();
  });
}, (req, res) => {
  res.json({ message: '上传成功', file: req.file });
});
```

## 完整示例：带验证的上传接口

```js
const express = require('express');
const multer = require('multer');
const path = require('path');
const fs = require('fs');

const app = express();

// 确保上传目录存在
const uploadDir = 'uploads/';
if (!fs.existsSync(uploadDir)) {
  fs.mkdirSync(uploadDir, { recursive: true });
}

// 精选存储
const storage = multer.diskStorage({
  destination(req, file, cb) {
    cb(null, uploadDir);
  },
  filename(req, file, cb) {
    const ext = path.extname(file.originalname).toLowerCase();
    cb(null, `${Date.now()}-${Math.round(Math.random() * 1e9)}${ext}`);
  },
});

// 上传配置
const upload = multer({
  storage,
  limits: { fileSize: 10 * 1024 * 1024 }, // 10 MB
  fileFilter(req, file, cb) {
    const allowed = ['.jpg', '.jpeg', '.png', '.gif', '.webp', '.pdf'];
    const ext = path.extname(file.originalname).toLowerCase();
    if (allowed.includes(ext)) {
      cb(null, true);
    } else {
      cb(new Error(`不支持的文件类型: ${ext}。支持: ${allowed.join(', ')}`));
    }
  },
});

// 上传接口
app.post('/upload', (req, res, next) => {
  upload.single('file')(req, res, (err) => {
    if (err) {
      const message = err instanceof multer.MulterError
        ? `上传错误 [${err.code}]: ${err.message}`
        : err.message;
      return res.status(400).json({ error: message });
    }
    next();
  });
}, (req, res) => {
  if (!req.file) {
    return res.status(400).json({ error: '请选择要上传的文件' });
  }
  res.json({
    message: '上传成功',
    file: {
      originalName: req.file.originalname,
      fileName: req.file.filename,
      size: req.file.size,
      url: `/uploads/${req.file.filename}`,
    },
  });
});

// 提供文件访问
app.use('/uploads', express.static(uploadDir));

app.listen(3000, () => console.log('Server on http://localhost:3000'));
```

## 常见问题

**1. 为什么 req.body 为空？** Multer 必须在使用 express.json() 之前注册。Multer 解析的是 multipart/form-data，express.json() 解析的是 application/json，两者互不冲突。

**2. 如何限制单个用户上传频率？** 结合 rate-limiter 中间件（如 express-rate-limit），按 IP 或用户 ID 限制上传端点访问频率。

**3. 大文件上传怎么办？** 超过 100MB 的文件建议用分片上传（chunked upload），前端分片、后端合并，避免单次请求超时。

## 总结

Multer 的配置要点：选择合适的存储方式（磁盘/内存），通过 fileFilter 限制文件类型，limits 限制大小，完整的错误处理覆盖 MulterError 和自定义错误。
