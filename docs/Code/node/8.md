# JWT 身份认证实战

JWT (JSON Web Token) 是无状态身份认证的主流方案。本文从用户注册到请求鉴权，实现完整的 JWT 认证流程。

## JWT 是什么

JWT 由三部分组成，用 `.` 连接：

```
header.payload.signature
```

- **Header**: 算法类型，`{"alg":"HS256","typ":"JWT"}`
- **Payload**: 声明数据，如 `{"userId":"123","iat":1700000000,"exp":1700604800}`
- **Signature**: 对前两部分的签名，防止篡改

JWT 使用 Base64URL 编码，可以在 jwt.io 在线解析验证。

## 实现流程

### 1. 用户注册

```js
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');

// 假设已有 users 数组作为简易数据存储
const users = [];

async function register(req, res) {
  const { username, password, email } = req.body;

  // 检查用户是否已存在
  if (users.find((u) => u.email === email)) {
    return res.status(409).json({ error: '该邮箱已注册' });
  }

  // 哈希密码
  const salt = await bcrypt.genSalt(10);
  const hashedPassword = await bcrypt.hash(password, salt);

  const user = {
    id: String(Date.now()),
    username,
    email,
    password: hashedPassword,
    createdAt: new Date(),
  };
  users.push(user);

  // 注册成功后签发 JWT
  const token = jwt.sign(
    { userId: user.id },
    process.env.JWT_SECRET,
    { expiresIn: '7d' },
  );

  res.status(201).json({
    message: '注册成功',
    token,
    user: { id: user.id, username: user.username, email: user.email },
  });
}
```

### 2. 用户登录

```js
async function login(req, res) {
  const { email, password } = req.body;

  const user = users.find((u) => u.email === email);
  if (!user) {
    return res.status(401).json({ error: '邮箱或密码错误' });
  }

  // 验证密码
  const isValid = await bcrypt.compare(password, user.password);
  if (!isValid) {
    return res.status(401).json({ error: '邮箱或密码错误' });
  }

  // 签发 access token（短期）
  const accessToken = jwt.sign(
    { userId: user.id },
    process.env.JWT_SECRET,
    { expiresIn: '15m' },
  );

  // 签发 refresh token（长期）
  const refreshToken = jwt.sign(
    { userId: user.id, type: 'refresh' },
    process.env.JWT_REFRESH_SECRET,
    { expiresIn: '7d' },
  );

  res.json({
    accessToken,
    refreshToken,
    user: { id: user.id, username: user.username, email: user.email },
  });
}
```

### 3. 认证中间件

```js
// middleware/auth.js
const jwt = require('jsonwebtoken');

function authenticate(req, res, next) {
  const authHeader = req.headers.authorization;
  if (!authHeader || !authHeader.startsWith('Bearer ')) {
    return res.status(401).json({ error: '未提供认证令牌' });
  }

  const token = authHeader.split(' ')[1];

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    req.userId = decoded.userId;
    next();
  } catch (err) {
    if (err.name === 'TokenExpiredError') {
      return res.status(401).json({ error: '令牌已过期，请刷新' });
    }
    return res.status(401).json({ error: '令牌无效' });
  }
}

module.exports = { authenticate };
```

### 4. 保护路由

```js
const { authenticate } = require('./middleware/auth');

// 公开路由
app.post('/api/register', register);
app.post('/api/login', login);

// 需要认证的路由
app.get('/api/profile', authenticate, (req, res) => {
  const user = users.find((u) => u.id === req.userId);
  if (!user) return res.status(404).json({ error: '用户不存在' });
  res.json({ id: user.id, username: user.username, email: user.email });
});

app.put('/api/profile', authenticate, (req, res) => {
  const user = users.find((u) => u.id === req.userId);
  if (!user) return res.status(404).json({ error: '用户不存在' });
  Object.assign(user, req.body, { updatedAt: new Date() });
  res.json({ message: '更新成功' });
});
```

## Token 刷新机制

```js
// routes/auth.js
async function refreshToken(req, res) {
  const { refreshToken } = req.body;
  if (!refreshToken) {
    return res.status(400).json({ error: '缺少 refreshToken' });
  }

  try {
    const decoded = jwt.verify(refreshToken, process.env.JWT_REFRESH_SECRET);
    if (decoded.type !== 'refresh') {
      return res.status(401).json({ error: '无效的刷新令牌' });
    }

    // 签发新的 access token
    const newAccessToken = jwt.sign(
      { userId: decoded.userId },
      process.env.JWT_SECRET,
      { expiresIn: '15m' },
    );

    res.json({ accessToken: newAccessToken });
  } catch (err) {
    return res.status(401).json({ error: '刷新令牌无效或已过期，请重新登录' });
  }
}
```

## 完整的认证流程路由

```js
const express = require('express');
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');
const { authenticate } = require('./middleware/auth');

const router = express.Router();
const users = [];

// 注册
router.post('/register', async (req, res) => {
  try {
    const { username, email, password } = req.body;

    if (users.find((u) => u.email === email)) {
      return res.status(409).json({ error: '该邮箱已注册' });
    }

    const hashedPassword = await bcrypt.hash(password, 10);
    const user = { id: String(Date.now()), username, email, password: hashedPassword };
    users.push(user);

    const token = jwt.sign({ userId: user.id }, process.env.JWT_SECRET, { expiresIn: '7d' });

    res.status(201).json({ token, user: { id: user.id, username, email } });
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
});

// 登录
router.post('/login', async (req, res) => {
  try {
    const { email, password } = req.body;
    const user = users.find((u) => u.email === email);
    if (!user) return res.status(401).json({ error: '邮箱或密码错误' });

    const valid = await bcrypt.compare(password, user.password);
    if (!valid) return res.status(401).json({ error: '邮箱或密码错误' });

    const accessToken = jwt.sign({ userId: user.id }, process.env.JWT_SECRET, { expiresIn: '15m' });
    const refreshToken = jwt.sign(
      { userId: user.id, type: 'refresh' },
      process.env.JWT_REFRESH_SECRET,
      { expiresIn: '7d' },
    );

    res.json({ accessToken, refreshToken });
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
});

// 获取当前用户
router.get('/me', authenticate, (req, res) => {
  const user = users.find((u) => u.id === req.userId);
  res.json({ id: user.id, username: user.username, email: user.email });
});

// 刷新 Token
router.post('/refresh', refreshToken);

module.exports = router;
```

## 安全要点

### 1. 短期 access token + 长期 refresh token

Access token 过期时间 15 分钟，泄露后影响窗口小。Refresh token 过期时间 7 天，用于获取新 access token。

### 2. 不要在 Payload 中存放敏感数据

```js
// ❌ 危险 — payload 是 base64 编码，不是加密，任何人可以解码
jwt.sign({ password: user.password }, secret);

// ✅ 只放非敏感标识信息
jwt.sign({ userId: user.id, role: user.role }, secret);
```

### 3. 使用足够强度的密钥

```bash
# 生成安全的密钥
node -e "console.log(require('crypto').randomBytes(64).toString('hex'))"
```

生产环境不要使用弱密钥（如 `'secret'`、`'123456'`）。

### 4. HttpOnly Cookie 存放 refresh token

```js
// refresh token 通过 httpOnly cookie 传递，防止 XSS 窃取
res.cookie('refreshToken', refreshToken, {
  httpOnly: true,
  secure: process.env.NODE_ENV === 'production',
  sameSite: 'strict',
  maxAge: 7 * 24 * 60 * 60 * 1000, // 7 天
  path: '/api/auth/refresh',
});
```

### 5. 登出场景处理

JWT 无法主动失效。常见的处理方式：
- 维护服务端黑名单（Redis 存储已失效 token ID）
- 缩短 access token 过期时间，依赖过期自动失效
- refresh token 轮换机制

## 总结

JWT 认证核心流程：注册/登录 -> 签发 token -> 客户端 Authorization 头携带 -> 中间件验证。安全上坚持：短期 access token、httpOnly cookie 存放 refresh token、强随机密钥、payload 不存敏感信息。
