# 数据库操作：Sequelize ORM 入门

Sequelize 是 Node.js 生态中最成熟的 ORM 之一，支持 MySQL、PostgreSQL、SQLite、MariaDB 和 MSSQL。本文以 PostgreSQL 为例讲解核心用法。

## 安装

```bash
npm install sequelize pg pg-hstore
# MySQL 用:
# npm install sequelize mysql2
```

## 创建 Sequelize 实例

```js
const { Sequelize } = require('sequelize');

const sequelize = new Sequelize('my_database', 'postgres', 'password', {
  host: 'localhost',
  dialect: 'postgres',
  port: 5432,
  logging: (msg) => console.log(msg), // 设为 false 关闭 SQL 日志
  pool: {
    max: 10,
    min: 0,
    acquire: 30000,
    idle: 10000,
  },
});

// 测试连接
async function testConnection() {
  try {
    await sequelize.authenticate();
    console.log('数据库连接成功');
  } catch (err) {
    console.error('连接失败:', err);
  }
}
```

## 定义模型

有三种方式定义模型，推荐使用 `sequelize.define()` 或 ES6 Class：

### define() 方式

```js
const User = sequelize.define('User', {
  id: {
    type: Sequelize.INTEGER,
    primaryKey: true,
    autoIncrement: true,
  },
  username: {
    type: Sequelize.STRING(50),
    allowNull: false,
    unique: true,
    validate: {
      len: [2, 50],
      notEmpty: true,
    },
  },
  email: {
    type: Sequelize.STRING(100),
    allowNull: false,
    unique: true,
    validate: {
      isEmail: true,
    },
  },
  age: {
    type: Sequelize.INTEGER,
    validate: {
      min: 0,
      max: 150,
    },
  },
  role: {
    type: Sequelize.ENUM('user', 'admin', 'moderator'),
    defaultValue: 'user',
  },
  isActive: {
    type: Sequelize.BOOLEAN,
    defaultValue: true,
  },
}, {
  tableName: 'users',       // 自定义表名
  timestamps: true,         // 自动管理 createdAt / updatedAt
  paranoid: false,          // true 启用软删除(需 deletedAt 字段)
  underscored: false,       // true 将 camelCase 转 snake_case
  indexes: [
    { fields: ['email'] },
    { fields: ['role'] },
  ],
});
```

## CRUD 操作

### 创建

```js
// 方式一：build + save
const user = User.build({ username: 'john', email: 'john@example.com' });
await user.save();

// 方式二：create（推荐）
const user = await User.create({
  username: 'john',
  email: 'john@example.com',
  age: 25,
});

// 批量创建
const users = await User.bulkCreate([
  { username: 'alice', email: 'alice@example.com' },
  { username: 'bob', email: 'bob@example.com' },
]);
```

### 查询

```js
// 查询所有
const users = await User.findAll();

// 按主键查询
const user = await User.findByPk(1);

// 查找单条
const user = await User.findOne({ where: { email: 'john@example.com' } });

// 带条件查询
const activeAdmins = await User.findAll({
  where: {
    role: 'admin',
    isActive: true,
    age: { [Op.gte]: 18 },  // Op 是 Sequelize.Op
  },
  order: [['createdAt', 'DESC']],
  limit: 10,
  offset: 0,
});

// 聚合查询
const count = await User.count({ where: { role: 'admin' } });
const maxAge = await User.max('age');
const minAge = await User.min('age');

// 查询特定字段
const usernames = await User.findAll({
  attributes: ['id', 'username', 'email'], // 只返回这些列
});
```

### 更新

```js
// 方式一：find + save
const user = await User.findByPk(1);
user.age = 30;
await user.save();

// 方式二：update 直接更新
await User.update(
  { age: 30, role: 'moderator' },
  { where: { id: 1 } },
);

// 批量更新
await User.update(
  { isActive: false },
  { where: { lastLoginAt: { [Op.lt]: new Date('2024-01-01') } } },
);
```

### 删除

```js
// 硬删除
await User.destroy({ where: { id: 1 } });

// 批量删除
await User.destroy({ where: { isActive: false } });

// 软删除（需在模型定义中设置 paranoid: true）
// 实际执行 UPDATE SET deletedAt = NOW()
await User.destroy({ where: { id: 1 } });
```

## 模型关联

```js
const { Sequelize, DataTypes } = require('sequelize');

// 定义模型
const User = sequelize.define('User', {
  id: { type: DataTypes.INTEGER, primaryKey: true, autoIncrement: true },
  username: { type: DataTypes.STRING, allowNull: false },
});

const Post = sequelize.define('Post', {
  id: { type: DataTypes.INTEGER, primaryKey: true, autoIncrement: true },
  title: { type: DataTypes.STRING, allowNull: false },
  content: { type: DataTypes.TEXT },
});

const Tag = sequelize.define('Tag', {
  id: { type: DataTypes.INTEGER, primaryKey: true, autoIncrement: true },
  name: { type: DataTypes.STRING, allowNull: false },
});

// 关联定义
User.hasMany(Post, { foreignKey: 'userId', as: 'posts' });
Post.belongsTo(User, { foreignKey: 'userId', as: 'author' });

Post.belongsToMany(Tag, { through: 'PostTags', as: 'tags' });
Tag.belongsToMany(Post, { through: 'PostTags', as: 'posts' });
```

### 关联查询 (Eager Loading)

```js
// 查询用户及其文章
const userWithPosts = await User.findByPk(1, {
  include: [{ model: Post, as: 'posts' }],
});

// 查询文章及其作者和标签
const post = await Post.findOne({
  where: { id: 1 },
  include: [
    { model: User, as: 'author', attributes: ['id', 'username'] },
    { model: Tag, as: 'tags', through: { attributes: [] } },
  ],
});

// 嵌套 include：用户 -> 文章 -> 标签
const user = await User.findByPk(1, {
  include: [{
    model: Post,
    as: 'posts',
    include: [{ model: Tag, as: 'tags' }],
  }],
});
```

### 关联创建

```js
// 创建用户的文章
const user = await User.findByPk(1);
const post = await user.createPost({
  title: '我的第一篇文章',
  content: 'Hello Sequelize!',
});

// 为文章添加标签
const tag = await Tag.create({ name: 'Node.js' });
await post.addTag(tag);
```

## 复杂查询条件

```js
const { Op } = Sequelize;

const posts = await Post.findAll({
  where: {
    [Op.or]: [
      { title: { [Op.like]: '%Node%' } },
      { content: { [Op.like]: '%Node%' } },
    ],
    status: {
      [Op.in]: ['published', 'draft'],
    },
    createdAt: {
      [Op.between]: [new Date('2024-01-01'), new Date('2024-12-31')],
    },
  },
  order: [
    ['createdAt', 'DESC'],
  ],
  limit: 20,
});
```

## 事务

```js
async function transferPoints(fromUserId, toUserId, amount) {
  const t = await sequelize.transaction();
  try {
    const fromUser = await User.findByPk(fromUserId, { transaction: t });
    const toUser = await User.findByPk(toUserId, { transaction: t });

    if (fromUser.points < amount) {
      throw new Error('积分不足');
    }

    fromUser.points -= amount;
    toUser.points += amount;

    await fromUser.save({ transaction: t });
    await toUser.save({ transaction: t });

    await t.commit();
    console.log('转账成功');
  } catch (err) {
    await t.rollback();
    console.error('转账失败，已回滚:', err.message);
    throw err;
  }
}
```

## 原生查询

ORM 无法处理的复杂查询，可以使用原生 SQL：

```js
// 查询
const [results] = await sequelize.query(
  'SELECT * FROM users WHERE age > :age',
  {
    replacements: { age: 18 },
    type: Sequelize.QueryTypes.SELECT,
  },
);

// 更新
await sequelize.query(
  'UPDATE users SET isActive = false WHERE lastLoginAt < :cutoff',
  { replacements: { cutoff: new Date('2024-01-01') } },
);
```

## 总结

Sequelize 的核心在于：模型定义、关联配置、CRUD API 和查询条件构建。它能显著减少手写 SQL，适合传统 MVC 架构项目。但注意 `include` 生成的 JOIN 查询在大数据量下可能存在 N+1 和性能问题，需要关注实际执行的 SQL。
