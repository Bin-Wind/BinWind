# 常用内置函数与标准库

Python "自备电池" 的哲学意味着标准库已经覆盖了大量常见需求。本节聚焦高频内置函数和几个核心标准库模块，让你的代码更简洁、更 Pythonic。

## 高频内置函数

```python
# map — 对每个元素应用函数
numbers = [1, 2, 3, 4, 5]
doubled = list(map(lambda x: x * 2, numbers))
print(doubled)                             # [2, 4, 6, 8, 10]

# filter — 筛选满足条件的元素
evens = list(filter(lambda x: x % 2 == 0, numbers))
print(evens)                               # [2, 4]

# zip — 并行打包多个可迭代对象
names = ["张三", "李四", "王五"]
scores = [92, 85, 78]
for name, score in zip(names, scores):
    print(f"{name}: {score}")

# 解压 zip
pairs = [("a", 1), ("b", 2), ("c", 3)]
letters, numbers = zip(*pairs)
print(letters)                             # ('a', 'b', 'c')

# enumerate — 带索引的遍历
for i, item in enumerate(["A", "B", "C"], start=1):
    print(f"{i}. {item}")

# sorted — 返回排序后的新列表
data = [3, 1, 4, 1, 5, 9]
print(sorted(data))                        # [1, 1, 3, 4, 5, 9]
print(sorted(data, reverse=True))          # [9, 5, 4, 3, 1, 1]

# 按自定义 key 排序
users = [("张三", 85), ("李四", 92), ("王五", 78)]
by_score = sorted(users, key=lambda u: u[1], reverse=True)
print(by_score)

# all / any — 全真 / 任一真
print(all([True, True, False]))            # False
print(any([False, False, True]))           # True
print(all(num > 0 for num in [1, 2, 3]))   # True

# max / min / sum
nums = [10, 5, 8, 12, 3]
print(max(nums), min(nums), sum(nums))     # 12 3 38
```

## collections - 高级容器

```python
from collections import (namedtuple, defaultdict,
                         OrderedDict, Counter, deque)

# namedtuple — 轻量级数据类
Point = namedtuple("Point", ["x", "y"])
p = Point(x=3, y=4)
print(p.x, p.y)          # 3 4
print(p[0], p[1])        # 3 4 — 同时支持索引和属性访问

# defaultdict — 带默认值的字典
word_count = defaultdict(int)
sentence = "to be or not to be"
for word in sentence.split():
    word_count[word] += 1                 # 不需要检查 key 是否存在
print(dict(word_count))                   # {'to': 2, 'be': 2, 'or': 1, 'not': 1}

# 使用 list 作为默认值实现分组
grouped = defaultdict(list)
students = [("A组", "张三"), ("B组", "李四"), ("A组", "王五")]
for group, name in students:
    grouped[group].append(name)
print(dict(grouped))                      # {'A组': ['张三', '王五'], 'B组': ['李四']}

# Counter — 计数器
colors = ["红", "蓝", "红", "绿", "蓝", "蓝"]
counter = Counter(colors)
print(counter)                            # Counter({'蓝': 3, '红': 2, '绿': 1})
print(counter.most_common(2))            # [('蓝', 3), ('红', 2)]

# deque — 高效双端队列
dq = deque([1, 2, 3])
dq.appendleft(0)                          # 左边追加
dq.append(4)                              # 右边追加
dq.popleft()                              # 左边弹出
dq.pop()                                  # 右边弹出
print(list(dq))                           # [1, 2, 3]
```

## datetime - 日期时间处理

```python
from datetime import datetime, date, timedelta, timezone

# 当前时间
now = datetime.now()
print(now)                                # 2024-06-15 14:30:22.123456

# 创建特定时间
dt = datetime(2024, 6, 15, 14, 30, 0)
print(dt)                                 # 2024-06-15 14:30:00

# 格式化输出
print(dt.strftime("%Y-%m-%d %H:%M:%S"))   # 2024-06-15 14:30:00
print(dt.strftime("%Y年%m月%d日"))         # 2024年06月15日

# 解析字符串
parsed = datetime.strptime("2024-06-15", "%Y-%m-%d")
print(parsed.date())                      # 2024-06-15

# 时间计算
tomorrow = now + timedelta(days=1)
last_week = now - timedelta(weeks=1)
diff = datetime(2024, 12, 31) - datetime(2024, 1, 1)
print(f"相差 {diff.days} 天")             # 相差 365 天

# 时间戳
timestamp = datetime.now().timestamp()
from_timestamp = datetime.fromtimestamp(timestamp)
```

## random - 随机数

```python
import random

# 基本随机
print(random.random())                    # [0.0, 1.0) 的浮点数
print(random.randint(1, 10))              # [1, 10] 的整数
print(random.uniform(1.0, 5.0))           # [1.0, 5.0] 的浮点数

# 从序列中选择
colors = ["红", "绿", "蓝", "黄"]
print(random.choice(colors))             # 随机选一个
print(random.sample(colors, k=2))        # 不重复地随机选 2 个

# 打乱顺序
deck = list(range(1, 53))
random.shuffle(deck)
print(deck[:5])                           # 前 5 张牌

# 设置随机种子 — 保证结果可复现
random.seed(42)
print(random.random())                    # 每次运行结果相同
```

## re - 正则表达式

```python
import re

text = "联系我：alice@example.com 或 bob@company.com"

# 搜索
match = re.search(r"(\w+)@(\w+\.\w+)", text)
print(match.group())                      # alice@example.com
print(match.group(1))                     # alice

# 查找所有匹配
emails = re.findall(r"\w+@\w+\.\w+", text)
print(emails)                             # ['alice@example.com', 'bob@company.com']

# 替换
masked = re.sub(r"(\w+)@", "***@", text)
print(masked)                             # 联系我：***@example.com 或 ***@company.com

# 编译模式（重复使用时更高效）
email_pattern = re.compile(r"(\w+)@(\w+\.\w+)")
print(email_pattern.findall(text))

# 分割
parts = re.split(r"\s+", "a  b   c    d")
print(parts)                              # ['a', 'b', 'c', 'd']
```

## os / os.path 常用操作

```python
import os

# 环境变量
home = os.environ.get("HOME")
path = os.environ.get("PATH")

# 路径操作（建议用 pathlib，这里展示和 pathlib 的对应关系）
print(os.path.join("dir", "subdir", "file.txt"))  # dir/subdir/file.txt
print(os.path.basename("/path/to/file.txt"))      # file.txt
print(os.path.dirname("/path/to/file.txt"))       # /path/to
print(os.path.splitext("file.txt"))               # ('file', '.txt')
print(os.path.exists("/path/to/check"))           # True/False

# 当前工作目录
cwd = os.getcwd()
print(cwd)

# 列出目录
entries = os.listdir(".")
print(entries[:5])                        # 前 5 个条目
```

> 掌握标准库，相当于拥有一个强大的工具箱。`collections` 解决数据结构问题，`datetime` 处理时间，`random` 生成测试数据，`re` 处理文本匹配——这些模块足以覆盖日常脚本 80% 的需求，不必每次都用第三方库。
