# 字符串处理与正则表达式

文本处理是编程中最常见的任务之一。Python 的字符串方法和正则表达式提供了从简单到复杂的全频谱文本操作能力。

## 字符串基础操作

```python
s = "  Hello, Python World!  "

# 大小写转换
print(s.upper())             # "  HELLO, PYTHON WORLD!  "
print(s.lower())             # "  hello, python world!  "
print(s.title())             # "  Hello, Python World!  "
print(s.capitalize())        # "  hello, python world!  "
print(s.swapcase())          # "  hELLO, pYTHON wORLD!  "

# 空白处理
print(s.strip())             # "Hello, Python World!"
print(s.lstrip())            # 去除左侧空白
print(s.rstrip())            # 去除右侧空白

# 前缀/后缀检查
print(s.strip().startswith("Hello"))      # True
print(s.strip().endswith("!"))           # True

# 查找与替换
text = "Python is great, Python is fun"
print(text.find("Python"))               # 0 — 第一次出现的位置
print(text.rfind("Python"))              # 17 — 最后一次出现的位置
print(text.count("Python"))              # 2 — 出现次数
print(text.replace("Python", "JS", 1))   # "JS is great, Python is fun" — 只替换第一个

# 判断字符类型
print("123".isdigit())                    # True
print("abc".isalpha())                    # True
print("abc123".isalnum())                 # True
print("   ".isspace())                    # True
print("Hello".islower())                  # False
print("HELLO".isupper())                  # True
```

## 字符串分割与拼接

```python
# split — 分割成列表
csv_line = "张三,28,北京,工程师"
fields = csv_line.split(",")
print(fields)                             # ['张三', '28', '北京', '工程师']

# 限制分割次数
print(csv_line.split(",", maxsplit=2))   # ['张三', '28', '北京,工程师']

# rsplit — 从右侧分割
print(csv_line.rsplit(",", maxsplit=1))  # ['张三,28,北京', '工程师']

# splitlines — 按换行符分割
multiline = "第一行\n第二行\r\n第三行"
print(multiline.splitlines())             # ['第一行', '第二行', '第三行']
print(multiline.splitlines(keepends=True))  # 保留换行符

# join — 列表拼接为字符串
words = ["Python", "is", "awesome"]
print(" ".join(words))                    # "Python is awesome"
print(", ".join(["A", "B", "C"]))         # "A, B, C"

# partition — 分割成三段（分隔符前、分隔符、分隔符后）
text = "name=alice&age=28"
left, sep, right = text.partition("=")
print(left, sep, right)                   # name = alice&age=28
```

## f-string 格式化

Python 3.6+ 引入的 f-string 是字符串格式化的最佳选择：

```python
name = "Alice"
age = 28
score = 92.567

# 基本插值
print(f"我叫 {name}，今年 {age} 岁")

# 表达式
print(f"明年 {age + 1} 岁")

# 格式化数字
print(f"成绩：{score:.2f}")               # 成绩：92.57
print(f"百分比：{0.8567:.1%}")            # 百分比：85.7%
print(f"金额：{1234567:,} 元")            # 金额：1,234,567 元

# 对齐与填充
print(f"|{'左对齐':<10}|")                # |左对齐      |
print(f"|{'右对齐':>10}|")                # |      右对齐|
print(f"|{'居中':^10}|")                  # |   居中   |
print(f"|{'填充':-^10}|")                 # |----填充----|

# 数字进制
num = 255
print(f"十进制: {num}, 十六进制: {num:#x}, 二进制: {num:#b}")
# 十进制: 255, 十六进制: 0xff, 二进制: 0b11111111

# 日期时间
from datetime import datetime
now = datetime.now()
print(f"当前时间：{now:%Y-%m-%d %H:%M:%S}")

# 多行 f-string
user = {"name": "Bob", "role": "Admin"}
msg = (
    f"用户名：{user['name']}\n"
    f"角色：{user['role']}"
)

# 调试模式 (Python 3.8+)
x = 10 + 20
print(f"{x=}")                            # x=30
print(f"{name.upper()=}")                 # name.upper()='ALICE'
```

## str.format() 补充用法

```python
# 位置参数
print("{} + {} = {}".format(1, 2, 1 + 2))

# 关键字参数
print("{name} 来自 {city}".format(name="张三", city="北京"))

# 索引与属性
data = ["Alice", 28]
print("{0[0]} 今年 {0[1]} 岁".format(data))

# 格式规范微语言
print("{:>10}".format("OK"))              # 右对齐，宽度 10
print("{:<10}".format("OK"))              # 左对齐，宽度 10
```

## 常用正则表达式 (re 模块)

```python
import re

# 预编译正则模式（复用场景，性能更好）
phone_pattern = re.compile(r"1[3-9]\d{9}")
email_pattern = re.compile(r"[\w.-]+@[\w-]+\.\w+")
id_pattern = re.compile(r"^\d{17}[\dXx]$")

text = "我的手机是13800138000，邮箱是test@example.com"

# search — 找到第一个匹配
match = phone_pattern.search(text)
if match:
    print(match.group())                  # 13800138000
    print(match.span())                   # (5, 16) — 匹配的起止位置

# findall — 找到所有匹配
numbers = re.findall(r"\d+", "a1b22c333")
print(numbers)                            # ['1', '22', '333']

# finditer — 返回迭代器（内存友好）
for m in re.finditer(r"\d+", "a1b22c333"):
    print(m.group(), m.start(), m.end())

# sub — 替换
text = "电话：13800138000 和 13900139000"
masked = re.sub(r"(\d{3})\d{4}(\d{4})", r"\1****\2", text)
print(masked)                             # 电话：138****8000 和 139****9000

# split — 按模式分割
parts = re.split(r"[,;\s]+", "a,b;c d")
print(parts)                              # ['a', 'b', 'c', 'd']
```

## 正则表达式常用模式速查

```python
# 字符类
r"\d"     # 数字 [0-9]
r"\w"     # 单词字符 [a-zA-Z0-9_]
r"\s"     # 空白字符 [ \t\n\r\f\v]
r"\D"     # 非数字
r"\W"     # 非单词字符
r"\S"     # 非空白字符
r"."      # 除换行外的任意字符

# 量词
r"a*"     # 0 次或多次
r"a+"     # 1 次或多次
r"a?"     # 0 次或 1 次
r"a{3}"   # 恰好 3 次
r"a{3,}"  # 至少 3 次
r"a{3,5}" # 3 到 5 次

# 边界
r"^"      # 字符串开头
r"$"      # 字符串结尾
r"\b"     # 单词边界
r"\B"     # 非单词边界

# 分组
r"(ab)+"  # 捕获组
r"(?:ab)+"# 非捕获组
r"(?P<name>...)"  # 命名组
```

## 实战：常用的文本处理模式

```python
import re

# 1. 提取 URL
text = "访问 https://example.com 或 http://test.org/path"
urls = re.findall(r"https?://[\w./-]+", text)
print(urls)    # ['https://example.com', 'http://test.org/path']

# 2. 验证邮政编码（中国）
def is_valid_zip(zip_code):
    return bool(re.fullmatch(r"\d{6}", zip_code))

print(is_valid_zip("100010"))    # True
print(is_valid_zip("10001"))     # False

# 3. 驼峰转下划线
def camel_to_snake(name):
    return re.sub(r"(?<!^)(?=[A-Z])", "_", name).lower()

print(camel_to_snake("camelCaseName"))   # camel_case_name
print(camel_to_snake("HTMLElement"))     # html_element

# 4. 清理 HTML 标签
html = "<p>Hello <b>World</b></p>"
clean = re.sub(r"<[^>]+>", "", html)
print(clean)                              # Hello World

# 5. 匹配中文
chinese_text = "联系电话：13800138000，联系人：张三"
chinese_chars = re.findall(r"[一-鿿]+", chinese_text)
print(chinese_chars)                      # ['联系电话', '联系人', '张三']
```

> f-string 是 Python 3.6+ 的字符串格式化首选方案——它比 `%` 格式化更可读，比 `str.format()` 更简洁。正则表达式适合处理复杂匹配、提取、替换场景，但不要用它解析 HTML/XML（使用专门的解析库如 BeautifulSoup）。对于简单的子串查找，`in` 运算符和 `str.find()` 比正则更快、更可读。
