## 数据筛选与过滤

```python
df = pd.DataFrame({
    "name": ["Alice", "Bob", "Charlie", "Diana", "Eve"],
    "age": [25, 30, 35, 28, 22],
    "city": ["Beijing", "Shanghai", "Beijing", "Guangzhou", "Beijing"],
    "salary": [15000, 22000, 18000, 20000, 12000],
})

# 选择列
print(df["name"])              # 单列，返回 Series
print(df[["name", "salary"]])  # 多列，返回 DataFrame

# 按条件过滤（布尔索引）
high_salary = df[df["salary"] > 18000]
print(high_salary)

# 组合条件
beijing_high = df[(df["city"] == "Beijing") & (df["salary"] > 15000)]
print(beijing_high)

# loc：按标签索引
print(df.loc[0, "name"])           # 单个值
print(df.loc[0:2, ["name", "age"]])  # 行切片 + 列选择

# iloc：按位置索引
print(df.iloc[0, 1])               # 第 0 行第 1 列
print(df.iloc[:3, [0, 3]])         # 前 3 行，第 0 和第 3 列

# query：SQL 风格筛选
result = df.query("salary > 15000 and city == 'Beijing'")
print(result)
```

## 数据清洗

```python
df = pd.DataFrame({
    "name": ["Alice", "Bob", "Charlie", None, "Eve"],
    "age": [25, -1, 35, 28, 22],
    "salary": [15000, 22000, None, 20000, 12000],
})

# 处理缺失值
df_filled = df.fillna({"name": "Unknown", "salary": 0})
df_dropped = df.dropna(subset=["name"])  # name 为空的行直接删除

# 处理异常值
df_valid = df[df["age"] > 0]  # 过滤掉 age 为负的脏数据
df.loc[df["age"] < 0, "age"] = np.nan  # 把异常值设为 NaN

# 重命名列
df = df.rename(columns={"name": "姓名", "salary": "薪资"})

# 添加新列
df["bonus"] = df["salary"] * 0.2
df["income_category"] = pd.cut(df["salary"],
    bins=[0, 15000, 20000, float("inf")],
    labels=["low", "medium", "high"])

# 删除列
df = df.drop(columns=["bonus"])
```

## groupby：分组聚合

groupby 是数据分析中最常用的操作之一，类似 SQL 的 GROUP BY：

```python
df = pd.DataFrame({
    "department": ["Tech", "Tech", "Sales", "Sales", "HR", "HR"],
    "employee": ["Alice", "Bob", "Charlie", "Diana", "Eve", "Frank"],
    "salary": [15000, 22000, 12000, 14000, 13000, 11000],
    "bonus": [3000, 4400, 2400, 2800, 2600, 2200],
})

# 按部门聚合
dept_stats = df.groupby("department").agg({
    "salary": ["mean", "sum", "count"],
    "bonus": "mean",
})
print(dept_stats)

# 常用聚合函数
summary = df.groupby("department")["salary"].agg(
    avg_salary="mean",
    total_salary="sum",
    employee_count="count",
    max_salary="max",
    min_salary="min",
)
print(summary.reset_index())  # 重置索引，让 department 变回列
```

## merge：关联合并

Pandas 提供了类似 SQL JOIN 的合并操作：

```python
employees = pd.DataFrame({
    "emp_id": [1, 2, 3, 4],
    "name": ["Alice", "Bob", "Charlie", "Diana"],
    "dept_id": [10, 10, 20, 30],
})

departments = pd.DataFrame({
    "dept_id": [10, 20, 30, 40],
    "dept_name": ["Tech", "Sales", "HR", "Finance"],
})

# 内连接（默认）
inner = employees.merge(departments, on="dept_id", how="inner")

# 左连接
left = employees.merge(departments, on="dept_id", how="left")

# 外连接
outer = employees.merge(departments, on="dept_id", how="outer")

print(left)
```

## 实战：销售数据分析报告

```python
import pandas as pd

# 模拟销售数据
data = {
    "date": pd.date_range("2024-01-01", periods=100, freq="D"),
    "product": np.random.choice(["Laptop", "Mouse", "Keyboard", "Monitor"], 100),
    "region": np.random.choice(["East", "West", "North", "South"], 100),
    "quantity": np.random.randint(1, 20, 100),
    "unit_price": np.random.uniform(10, 1000, 100).round(2),
}
sales = pd.DataFrame(data)
sales["revenue"] = sales["quantity"] * sales["unit_price"]
sales["month"] = sales["date"].dt.to_period("M")  # 提取月份

# 月度营收汇总
monthly = sales.groupby("month")["revenue"].sum()
print("月度营收:\n", monthly)

# 各产品在各大区的销售额
pivot = sales.pivot_table(
    values="revenue",
    index="product",
    columns="region",
    aggfunc="sum",
    fill_value=0,
)
print("产品-区域交叉表:\n", pivot)

# 导出报表
with pd.ExcelWriter("sales_report.xlsx") as writer:
    sales.to_excel(writer, sheet_name="Raw Data", index=False)
    monthly.to_excel(writer, sheet_name="Monthly Summary")
    pivot.to_excel(writer, sheet_name="Pivot Table")
```

Pandas 的功能远不止于此：时间序列分析、窗口函数、链式操作（method chaining）等都是进阶主题。但掌握本文的 DataFrame 增删改查 + groupby + merge，你已经能应对 80% 的数据分析任务。
