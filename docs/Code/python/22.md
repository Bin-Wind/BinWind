# NumPy 数值计算基础

NumPy（Numerical Python）是 Python 科学计算的核心库，几乎所有数据科学库（Pandas、SciPy、scikit-learn）都构建在它之上。它的核心是 `ndarray` —— 一种高性能的多维数组对象，比 Python 原生列表快 10-100 倍。

## ndarray：Python 数值计算的主力

```python
import numpy as np

# 从列表创建
arr = np.array([1, 2, 3, 4, 5])
print(arr)          # [1 2 3 4 5]
print(type(arr))    # <class 'numpy.ndarray'>
print(arr.dtype)    # int64
print(arr.shape)    # (5,)
print(arr.ndim)     # 1 (维度数)
print(arr.size)     # 5 (元素总数)

# 创建特殊数组
zeros = np.zeros((3, 4))              # 3x4 全 0
ones = np.ones((2, 3))                # 2x3 全 1
eye = np.eye(4)                       # 4x4 单位矩阵
full = np.full((2, 2), 7)             # 2x2 全 7
arange = np.arange(0, 10, 2)          # [0 2 4 6 8]
linspace = np.linspace(0, 1, 5)       # [0.  0.25 0.5  0.75 1. ]

# 随机数数组
rand = np.random.rand(3, 2)           # 3x2, [0,1) 均匀分布
randn = np.random.randn(3, 2)         # 3x2, 标准正态分布
randint = np.random.randint(1, 100, (3, 2))  # 3x2, [1,100) 整数

# 查看与修改形状
arr = np.arange(12)
matrix = arr.reshape(3, 4)  # 3x4 矩阵
flattened = matrix.flatten()  # 展平为一维
print(matrix)
```

## 索引与切片

```python
arr = np.arange(24).reshape(2, 3, 4)
print(arr.shape)  # (2, 3, 4) — 2 层，3 行，4 列

# 基本索引
print(arr[0, 1, 2])       # 第 0 层、第 1 行、第 2 列

# 切片
print(arr[0, :, :])       # 第 0 层的全部行和列
print(arr[:, 0, :])       # 所有层的第 0 行
print(arr[0, 1:3, ::2])   # 第 0 层，第 1-2 行，每 2 列取一个

# 布尔索引（条件筛选）
data = np.array([15, 22, 8, 31, 5, 44])
print(data[data > 20])                  # [22 31 44]
print(data[(data > 10) & (data < 40)])  # [15 22 31]

# 花式索引（用整数数组）
indices = [0, 2, 4]
print(data[indices])  # [15 8 5]
```

## Broadcasting：不同形状数组的运算

Broadcasting 是 NumPy 最强大的特性之一，它允许不同形状的数组进行算术运算，无需显式复制数据：

```python
# 标量与数组
arr = np.array([1, 2, 3, 4])
print(arr * 10)       # [10 20 30 40]
print(arr + 100)      # [101 102 103 104]

# 不同形状数组相加
matrix = np.array([[10, 20, 30],
                   [40, 50, 60]])       # shape (2, 3)
row_vec = np.array([1, 2, 3])           # shape (3,)
result = matrix + row_vec               # row_vec 被 broadcast 到 (2, 3)
print(result)
# [[11 22 33]
#  [41 52 63]]

col_vec = np.array([[100], [200]])      # shape (2, 1)
result = matrix + col_vec               # col_vec 被 broadcast 到 (2, 3)
print(result)
# [[110 120 130]
#  [240 250 260]]

# Broadcasting 规则：
# 1. 从末尾维度开始对齐
# 2. 维度相同或其中一个为 1 即可运算
# 3. 缺失的维度视为 1
```

## ufunc：通用函数

ufunc（Universal Function）是 NumPy 的核心抽象，对数组进行逐元素运算，速度远快于 Python 循环：

```python
arr = np.array([1, 4, 9, 16, 25])

# 数学函数
print(np.sqrt(arr))       # [1. 2. 3. 4. 5.]
print(np.log(arr))        # 自然对数
print(np.exp(arr))        # e^x
print(np.sin(arr))        # 三角函数

# 聚合函数
print(np.sum(arr))        # 55
print(np.mean(arr))       # 11.0
print(np.median(arr))     # 9.0
print(np.std(arr))        # 标准差
print(np.min(arr), np.max(arr))  # 1 25

# 沿轴聚合
matrix = np.array([[1, 2, 3],
                   [4, 5, 6]])
print(np.sum(matrix, axis=0))  # [5 7 9] — 按列求和
print(np.sum(matrix, axis=1))  # [6 15]  — 按行求和

# 向量化操作比循环快 50-100 倍
```

## 线性代数基础

```python
# 矩阵乘法
a = np.array([[1, 2], [3, 4]])
b = np.array([[5, 6], [7, 8]])
print(a @ b)              # Python 3.5+ 矩阵乘法运算符
print(np.dot(a, b))       # 等价写法
print(np.matmul(a, b))    # 等价写法

# 转置
print(a.T)                # [[1 3], [2 4]]

# 逆矩阵
inv_a = np.linalg.inv(a)
print(inv_a)              # [[-2.  1. ], [ 1.5 -0.5]]
print(a @ inv_a)          # 接近单位矩阵

# 行列式 / 特征值
print(np.linalg.det(a))                     # -2.0
eigenvalues, eigenvectors = np.linalg.eig(a)
print("特征值:", eigenvalues)

# 解线性方程组: Ax = b
A = np.array([[3, 1], [1, 2]])
b_vec = np.array([9, 8])
x = np.linalg.solve(A, b_vec)
print(x)  # [2. 3.] — 验证: 3*2+3=9, 2+2*3=8
```

## 性能对比：NumPy vs 原生 Python

```python
import timeit

size = 1_000_000

# 原生 Python
def python_sum():
    return sum([i ** 2 for i in range(size)])

# NumPy
def numpy_sum():
    return np.sum(np.arange(size) ** 2)

# 实测 NumPy 快 10-30 倍
print("Python:", timeit.timeit(python_sum, number=10))
print("NumPy:", timeit.timeit(numpy_sum, number=10))
```

NumPy 快的原因有三：(1) 底层用 C/Fortran 实现，编译为机器码；(2) 数据连续存储，CPU 缓存命中率高；(3) 向量化运算消除 Python 循环和函数调用开销。

## 常用技巧

```python
# 条件选择 (类似 Excel 的 IF)
arr = np.array([15, 22, 8, 31, 5])
result = np.where(arr > 10, arr, -1)  # [15 22 -1 31 -1]

# 去重与集合操作
arr = np.array([1, 2, 2, 3, 3, 3])
print(np.unique(arr))                     # [1 2 3]
print(np.intersect1d([1,3,5], [2,3,4]))   # [3]
print(np.union1d([1,3,5], [2,3,4]))       # [1 2 3 4 5]

# 数据保存与加载
np.save("data.npy", arr)                  # 二进制格式
loaded = np.load("data.npy")
np.savetxt("data.csv", arr, delimiter=",")
loaded_csv = np.loadtxt("data.csv", delimiter=",")

# NaN 处理
data = np.array([1.0, np.nan, 3.0, np.nan, 5.0])
print(np.isnan(data))                     # [False True False True False]
print(np.nanmean(data))                   # 3.0（忽略 NaN 的均值）

# 内存视图 (view) vs 复制 (copy)
arr = np.array([1, 2, 3, 4, 5])
view = arr[1:4]       # view，共享内存
copy = arr[1:4].copy()  # 独立副本
view[0] = 999
print(arr)            # [1 999 3 4 5] — 原数组被修改了
```
