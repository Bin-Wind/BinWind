# Flask Web 框架入门

Flask 是 Python 最流行的轻量级 Web 框架，"micro" 的设计哲学意味着它只提供核心功能（路由、模板、请求处理），其余能力通过扩展按需添加。这种极简设计让 Flask 成为学习 Web 开发和小型项目的理想选择。

## 环境准备与第一个应用

```bash
pip install flask
```

```python
# app.py
from flask import Flask

app = Flask(__name__)

@app.route("/")
def home():
    return "<h1>Hello, Flask!</h1>"

@app.route("/api/health")
def health():
    return {"status": "ok", "version": "1.0"}

if __name__ == "__main__":
    app.run(debug=True, port=5000)
```

运行 `python app.py`，打开 `http://127.0.0.1:5000` 即可看到结果。`debug=True` 启用热重载和改进的错误页面。

## 路由与变量规则

```python
# 静态路由
@app.route("/about")
def about():
    return "About page"

# 动态路由（URL 变量）
@app.route("/user/<username>")
def profile(username: str):
    return f"<h1>User: {username}</h1>"

# 类型限定（string / int / float / path / uuid）
@app.route("/post/<int:post_id>")
def show_post(post_id: int):
    return f"<h1>Post #{post_id}</h1>"

# path 类型接受斜杠
@app.route("/files/<path:filepath>")
def serve_file(filepath: str):
    return f"File: {filepath}"

# 多个路由指向同一视图
@app.route("/home")
@app.route("/index")
def index():
    return "Welcome!"
```

## 请求处理

```python
from flask import request, abort

@app.route("/search")
def search():
    # URL 查询参数: /search?q=python&page=2
    q = request.args.get("q", "")
    page = request.args.get("page", 1, type=int)
    return f"Searching: {q}, Page: {page}"

@app.route("/login", methods=["GET", "POST"])
def login():
    if request.method == "POST":
        username = request.form.get("username")
        password = request.form.get("password")
        if username == "admin" and password == "secret":
            return "Login success"
        abort(401)  # 返回 401 Unauthorized
    return _render_login_form()

@app.route("/api/data", methods=["POST"])
def receive_data():
    data = request.get_json()  # 解析 JSON 请求体
    if not data:
        abort(400)
    return {"received": data, "count": len(data)}, 201
```

## Jinja2 模板渲染

Flask 默认使用 Jinja2 模板引擎。模板文件放在 `templates/` 目录下。

```html
<!-- templates/base.html -->
<!DOCTYPE html>
<html lang="zh">
<head>
    <meta charset="UTF-8">
    <title>{% block title %}My App{% endblock %}</title>
</head>
<body>
    <nav>
        <a href="{{ url_for('home') }}">Home</a>
    </nav>
    {% block content %}{% endblock %}
</body>
</html>
```

```html
<!-- templates/user.html -->
{% extends "base.html" %}
{% block title %}{{ user.name }}'s Profile{% endblock %}
{% block content %}
    <h1>{{ user.name }}</h1>
    <p>Age: {{ user.age }}</p>
    <ul>
    {% for item in items %}
        <li>{{ loop.index }}. {{ item }}</li>
    {% endfor %}
    </ul>
{% endblock %}
```

```python
from flask import render_template

@app.route("/user/<name>")
def user_page(name: str):
    context = {
        "user": {"name": name, "age": 30},
        "items": ["Python", "Flask", "Jinja2"],
    }
    return render_template("user.html", **context)

# url_for 生成 URL
@app.route("/test-url")
def test_url():
    return {
        "home": url_for("home"),
        "profile": url_for("profile", username="alice"),
        "show_post": url_for("show_post", post_id=42),
    }
```

## 表单处理与 Redirect / Flash

```python
from flask import request, redirect, url_for, flash

@app.route("/register", methods=["GET", "POST"])
def register():
    if request.method == "POST":
        email = request.form.get("email", "").strip()
        password = request.form.get("password", "")

        errors = []
        if not email or "@" not in email:
            errors.append("Invalid email")
        if len(password) < 6:
            errors.append("Password too short")

        if errors:
            for error in errors:
                flash(error, "error")
            return render_template("register.html")

        # 注册成功，重定向到首页
        flash("Registration successful!", "success")
        return redirect(url_for("home"))

    return render_template("register.html")
```

模板中显示 Flash 消息（`templates/base.html`）：

```html
{% with messages = get_flashed_messages(with_categories=true) %}
    {% if messages %}
        {% for category, message in messages %}
            <div class="flash {{ category }}">{{ message }}</div>
        {% endfor %}
    {% endif %}
{% endwith %}
```

## Blueprint：模块化应用

Blueprint 让你将应用拆分为可复用的模块：

```python
# auth.py — 认证模块
from flask import Blueprint, request, jsonify

auth_bp = Blueprint("auth", __name__, url_prefix="/auth")

@auth_bp.route("/register", methods=["POST"])
def register():
    return jsonify({"message": "Registered"})

@auth_bp.route("/login", methods=["POST"])
def login():
    return jsonify({"token": "abc123"})
```

```python
# blog.py — 博客模块
from flask import Blueprint

blog_bp = Blueprint("blog", __name__, url_prefix="/blog")

@blog_bp.route("/")
def list_posts():
    return {"posts": []}

@blog_bp.route("/<int:post_id>")
def view_post(post_id: int):
    return {"post": {"id": post_id, "title": "Hello"}}
```

```python
# app.py — 注册所有 Blueprint
from flask import Flask
from auth import auth_bp
from blog import blog_bp

app = Flask(__name__)
app.secret_key = "your-secret-key-here"
app.register_blueprint(auth_bp)
app.register_blueprint(blog_bp)
```

## 构建一个简单的 REST API

```python
from flask import Flask, request, jsonify, abort

app = Flask(__name__)

# 模拟数据库
tasks = [
    {"id": 1, "title": "Learn Flask", "done": False},
    {"id": 2, "title": "Build API", "done": False},
]
next_id = 3

@app.get("/api/tasks")
def list_tasks():
    done = request.args.get("done")
    if done is not None:
        is_done = done.lower() == "true"
        return jsonify([t for t in tasks if t["done"] == is_done])
    return jsonify(tasks)

@app.get("/api/tasks/<int:task_id>")
def get_task(task_id: int):
    task = next((t for t in tasks if t["id"] == task_id), None)
    if task is None:
        abort(404)
    return jsonify(task)

@app.post("/api/tasks")
def create_task():
    global next_id
    data = request.get_json()
    if not data or "title" not in data:
        abort(400, "Title is required")
    task = {"id": next_id, "title": data["title"], "done": False}
    next_id += 1
    tasks.append(task)
    return jsonify(task), 201

@app.put("/api/tasks/<int:task_id>")
def update_task(task_id: int):
    task = next((t for t in tasks if t["id"] == task_id), None)
    if task is None:
        abort(404)
    data = request.get_json()
    task["title"] = data.get("title", task["title"])
    task["done"] = data.get("done", task["done"])
    return jsonify(task)

@app.delete("/api/tasks/<int:task_id>")
def delete_task(task_id: int):
    global tasks
    task = next((t for t in tasks if t["id"] == task_id), None)
    if task is None:
        abort(404)
    tasks = [t for t in tasks if t["id"] != task_id]
    return "", 204
```

Flask 的核心理念是"简单但不廉价"——它给你足够的工具快速搭建应用，同时保留了按需扩展的灵活性。对于小型 API、管理后台和个人项目，Flask 是不二之选。对于需要高性能、自动文档和类型验证的 API 项目，下一章的 FastAPI 可能更适合你。
