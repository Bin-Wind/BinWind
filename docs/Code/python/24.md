# FastAPI 搭建高性能 API

FastAPI 是近年来增长最快的 Python Web 框架之一。它基于 Starlette（ASGI）和 Pydantic（数据验证），原生支持 async/await、自动生成 OpenAPI 文档、依赖注入和类型驱动的请求验证。如果说 Flask 是轻巧的瑞士军刀，FastAPI 就是为高性能 API 而生的精密工具。

## 环境准备与第一个 API

```bash
pip install "fastapi[standard]"
```

```python
# main.py
from fastapi import FastAPI
from datetime import datetime

app = FastAPI(title="My API", version="1.0.0")

@app.get("/")
async def root():
    return {"message": "Hello, FastAPI!", "timestamp": datetime.now().isoformat()}

@app.get("/health")
async def health_check():
    return {"status": "healthy"}
```

启动：

```bash
fastapi dev main.py          # 开发模式（热重载）
# 或
uvicorn main:app --reload    # 等价命令

# 访问:
# API:        http://127.0.0.1:8000
# Swagger UI: http://127.0.0.1:8000/docs
# ReDoc:      http://127.0.0.1:8000/redoc
```

FastAPI 自动为你的 API 生成交互式文档（Swagger UI），你可以直接在浏览器中测试所有端点——这是它最受欢迎的杀手级特性之一。

## 路径参数、查询参数与请求体

```python
from fastapi import FastAPI, Query, Path, Body
from pydantic import BaseModel, Field
from enum import Enum
from typing import Annotated

app = FastAPI()

class Category(str, Enum):
    TECH = "tech"
    LIFE = "life"
    SPORTS = "sports"

class ArticleCreate(BaseModel):
    title: str = Field(min_length=1, max_length=200, description="文章标题")
    content: str = Field(min_length=10, description="文章内容")
    category: Category
    tags: list[str] = Field(default_factory=list, max_length=10)
    published: bool = False

# 路径参数
@app.get("/articles/{article_id}")
async def get_article(
    article_id: int = Path(ge=1, description="文章 ID"),
):
    return {"id": article_id, "title": "Sample Article"}

# 查询参数
@app.get("/articles")
async def list_articles(
    page: int = Query(default=1, ge=1),
    size: int = Query(default=20, ge=1, le=100),
    category: Category | None = None,
    keyword: str | None = Query(default=None, min_length=2),
):
    return {
        "page": page,
        "size": size,
        "category": category,
        "keyword": keyword,
    }

# POST 请求体
@app.post("/articles", status_code=201)
async def create_article(article: ArticleCreate):
    return {"message": "Article created", "data": article.model_dump()}

# 混合：路径 + 查询 + 请求体
@app.put("/articles/{article_id}")
async def update_article(
    article_id: int,
    article: ArticleCreate,
    force: bool = Query(default=False),
):
    return {
        "id": article_id,
        "force": force,
        "updates": article.model_dump(exclude_unset=True),
    }
```

## Pydantic 模型进阶

```python
from pydantic import BaseModel, Field, EmailStr, field_validator
from typing import Optional
from datetime import date

class UserCreate(BaseModel):
    username: str = Field(min_length=3, max_length=50, pattern=r"^[a-zA-Z0-9_]+$")
    email: EmailStr  # 自动验证 email 格式（需 pip install email-validator）
    age: int = Field(ge=0, le=150)
    bio: str | None = Field(default=None, max_length=500)
    join_date: date

    @field_validator("username")
    @classmethod
    def username_must_not_be_reserved(cls, v: str) -> str:
        reserved = {"admin", "root", "system"}
        if v.lower() in reserved:
            raise ValueError(f"Username '{v}' is reserved")
        return v

    model_config = {
        "json_schema_extra": {
            "example": {
                "username": "alice",
                "email": "alice@example.com",
                "age": 28,
                "join_date": "2024-01-15",
            }
        }
    }
```

## Dependency Injection：依赖注入

FastAPI 的依赖注入系统让代码复用和分离关注点变得异常优雅：

```python
from fastapi import Depends, HTTPException, Header
from typing import Annotated

# 简单依赖：共享查询参数
async def common_params(
    page: int = Query(default=1, ge=1),
    size: int = Query(default=20, ge=1, le=100),
):
    return {"page": page, "size": size}

CommonParams = Annotated[dict, Depends(common_params)]

@app.get("/items")
async def list_items(params: CommonParams):
    return params

# 认证依赖
def verify_token(authorization: str = Header()) -> dict:
    if not authorization.startswith("Bearer "):
        raise HTTPException(status_code=401, detail="Invalid token")
    token = authorization.removeprefix("Bearer ")
    # 实际项目中应验证 JWT 或查询数据库
    if token != "secret-token":
        raise HTTPException(status_code=403, detail="Invalid credentials")
    return {"user_id": 1, "username": "alice"}

CurrentUser = Annotated[dict, Depends(verify_token)]

@app.get("/me")
async def get_current_user(user: CurrentUser):
    return {"user": user}

# 数据库会话依赖（实际项目常用模式）
from typing import Generator

def get_db() -> Generator:
    db = "fake-db-connection"
    try:
        yield db
    finally:
        pass  # 关闭数据库连接

DbSession = Annotated[str, Depends(get_db)]

@app.get("/db-test")
async def db_test(db: DbSession):
    return {"db": db}
```

## 错误处理与中间件

```python
from fastapi import Request
from fastapi.responses import JSONResponse
import time

# 全局异常处理
@app.exception_handler(ValueError)
async def value_error_handler(request: Request, exc: ValueError):
    return JSONResponse(
        status_code=422,
        content={"detail": str(exc)},
    )

# CORS 中间件
from fastapi.middleware.cors import CORSMiddleware

app.add_middleware(
    CORSMiddleware,
    allow_origins=["http://localhost:3000"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# 自定义中间件（请求计时）
@app.middleware("http")
async def add_process_time_header(request: Request, call_next):
    start = time.perf_counter()
    response = await call_next(request)
    process_time = time.perf_counter() - start
    response.headers["X-Process-Time"] = f"{process_time:.4f}s"
    return response
```

## 文件上传与静态文件

```python
from fastapi import UploadFile, File
from fastapi.staticfiles import StaticFiles
import shutil
import os

# 单文件上传
@app.post("/upload")
async def upload_file(file: UploadFile = File(...)):
    contents = await file.read()
    file_size = len(contents)
    # 保存文件
    path = f"uploads/{file.filename}"
    os.makedirs("uploads", exist_ok=True)
    with open(path, "wb") as f:
        f.write(contents)
    return {
        "filename": file.filename,
        "content_type": file.content_type,
        "size": file_size,
    }

# 多文件上传
@app.post("/upload-multiple")
async def upload_multiple(files: list[UploadFile]):
    results = []
    for file in files:
        contents = await file.read()
        results.append({"filename": file.filename, "size": len(contents)})
    return {"files": results}

# 挂载静态文件目录
app.mount("/static", StaticFiles(directory="static"), name="static")
```

## 完整实战：Bookmarks API

```python
from fastapi import FastAPI, HTTPException, Depends, Query
from pydantic import BaseModel
from typing import Annotated
from datetime import datetime

app = FastAPI(title="Bookmarks API", version="1.0")

# --- Models ---
class BookmarkCreate(BaseModel):
    url: str
    title: str
    tags: list[str] = []

class Bookmark(BaseModel):
    id: int
    url: str
    title: str
    tags: list[str]
    created_at: datetime

# --- In-memory storage ---
bookmarks: dict[int, Bookmark] = {}
next_id = 1

# --- CRUD ---
@app.get("/bookmarks", response_model=list[Bookmark])
async def list_bookmarks(
    tag: str | None = Query(default=None),
    search: str | None = Query(default=None),
):
    result = list(bookmarks.values())
    if tag:
        result = [b for b in result if tag in b.tags]
    if search:
        result = [b for b in result if search.lower() in b.title.lower()]
    return result

@app.get("/bookmarks/{bookmark_id}", response_model=Bookmark)
async def get_bookmark(bookmark_id: int):
    if bookmark_id not in bookmarks:
        raise HTTPException(status_code=404, detail="Bookmark not found")
    return bookmarks[bookmark_id]

@app.post("/bookmarks", response_model=Bookmark, status_code=201)
async def create_bookmark(data: BookmarkCreate):
    global next_id
    bookmark = Bookmark(
        id=next_id,
        url=data.url,
        title=data.title,
        tags=data.tags,
        created_at=datetime.now(),
    )
    bookmarks[next_id] = bookmark
    next_id += 1
    return bookmark

@app.delete("/bookmarks/{bookmark_id}", status_code=204)
async def delete_bookmark(bookmark_id: int):
    if bookmark_id not in bookmarks:
        raise HTTPException(status_code=404, detail="Bookmark not found")
    del bookmarks[bookmark_id]
```

FastAPI 将 Python 类型系统的优势发挥到了极致——类型注解既是数据验证，也是 API 文档，还是编辑器自动补全的来源。对于中大型 API 项目、微服务架构和需要自动文档的团队开发，FastAPI 是当下最优选择。
