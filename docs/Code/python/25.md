## 场景一：批量文件重命名与整理

需求：将下载目录中的文件按扩展名分类到子文件夹，并统一重命名格式。

```python
import os
import shutil
from pathlib import Path
from datetime import datetime

def organize_and_rename(directory: str) -> None:
    """按扩展名分类文件并添加日期前缀"""
    target = Path(directory).expanduser()
    if not target.exists():
        print(f"目录不存在: {target}")
        return

    today = datetime.now().strftime("%Y%m%d")
    stats = {"moved": 0, "renamed": 0, "skipped": 0}

    for file_path in target.iterdir():
        if not file_path.is_file() or file_path.name.startswith("."):
            continue

        ext = file_path.suffix.lower().lstrip(".") or "noext"
        category_dir = target / ext
        category_dir.mkdir(exist_ok=True)

        # 新文件名: 日期_原名
        new_name = f"{today}_{file_path.name}"
        new_path = category_dir / new_name

        try:
            shutil.move(str(file_path), str(new_path))
            stats["moved"] += 1
            print(f"  MOVED: {file_path.name} -> {ext}/{new_name}")
        except Exception as e:
            stats["skipped"] += 1
            print(f"  ERROR: {file_path.name}: {e}")

    print(f"\n完成: {stats['moved']} 个文件已整理, {stats['skipped']} 个跳过")

# organize_and_rename("~/Downloads/test_folder")
```

进阶：按文件名模式批量重命名——比如将 `IMG_001.jpg` 改为 `Vacation_2024_001.jpg`：

```python
import re

def batch_rename(directory: str, pattern: str, replacement: str) -> int:
    """正则批量重命名文件"""
    dir_path = Path(directory).expanduser()
    count = 0

    for file_path in dir_path.iterdir():
        if not file_path.is_file():
            continue
        new_name = re.sub(pattern, replacement, file_path.name)
        if new_name != file_path.name:
            new_path = file_path.with_name(new_name)
            file_path.rename(new_path)
            print(f"  {file_path.name} -> {new_name}")
            count += 1

    return count
```

## 场景二：Excel 报表自动化生成

需求：读取销售数据 CSV，按产品汇总，生成带图表的 Excel 报表。

```python
import pandas as pd
from pathlib import Path
from datetime import datetime

def generate_sales_report(
    data_csv: str,
    output_dir: str = "reports",
) -> str:
    """从 CSV 生成格式化 Excel 销售报表"""
    output_path = Path(output_dir)
    output_path.mkdir(exist_ok=True)

    # 1. 加载数据
    df = pd.read_csv(data_csv, parse_dates=["date"])
    print(f"已加载 {len(df)} 条销售记录")

    # 2. 数据清洗
    df["revenue"] = df["quantity"] * df["unit_price"]
    df["month"] = df["date"].dt.strftime("%Y-%m")

    # 3. 维度汇总
    by_product = (
        df.groupby("product")
        .agg(
            销量=("quantity", "sum"),
            营收=("revenue", "sum"),
            均价=("unit_price", "mean"),
            订单数=("quantity", "count"),
        )
        .round(2)
    )

    by_month = (
        df.groupby("month")
        .agg(月营收=("revenue", "sum"), 月销量=("quantity", "sum"))
        .round(2)
        .sort_index()
    )

    top_products = by_product.sort_values("营收", ascending=False).head(10)

    # 4. 写入 Excel（多个 Sheet + 格式）
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    output_file = output_path / f"sales_report_{timestamp}.xlsx"

    with pd.ExcelWriter(output_file, engine="openpyxl") as writer:
        df.to_excel(writer, sheet_name="原始数据", index=False)
        by_product.to_excel(writer, sheet_name="产品汇总")
        by_month.to_excel(writer, sheet_name="月度趋势")
        top_products.to_excel(writer, sheet_name="Top10产品")

    print(f"报表已生成: {output_file}")
    return str(output_file)

# 生成模拟数据并测试
```

## 场景三：邮件自动发送

```python
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
from email.mime.application import MIMEApplication
from pathlib import Path

def send_email(
    smtp_server: str,
    smtp_port: int,
    sender: str,
    password: str,
    recipients: list[str],
    subject: str,
    body: str,
    attachments: list[str] | None = None,
    use_tls: bool = True,
) -> bool:
    """发送带附件的邮件"""
    msg = MIMEMultipart()
    msg["From"] = sender
    msg["To"] = ", ".join(recipients)
    msg["Subject"] = subject
    msg.attach(MIMEText(body, "plain", "utf-8"))

    # 添加附件
    if attachments:
        for filepath in attachments:
            path = Path(filepath)
            if not path.exists():
                print(f"附件不存在: {filepath}")
                continue
            with open(path, "rb") as f:
                part = MIMEApplication(f.read(), name=path.name)
                part["Content-Disposition"] = f'attachment; filename="{path.name}"'
                msg.attach(part)

    try:
        if use_tls:
            server = smtplib.SMTP(smtp_server, smtp_port, timeout=15)
            server.starttls()
        else:
            server = smtplib.SMTP_SSL(smtp_server, smtp_port, timeout=15)

        server.login(sender, password)
        server.sendmail(sender, recipients, msg.as_string())
        server.quit()
        print(f"邮件已发送至 {len(recipients)} 个收件人")
        return True
    except Exception as e:
        print(f"邮件发送失败: {e}")
        return False
```

## 最佳实践

1. **使用 `pathlib` 而非 `os.path`** — 跨平台、面向对象的路径操作更安全。
2. **异常处理要具体** — 捕获 `FileNotFoundError`、`smtplib.SMTPException` 等具体异常，而非裸 `except`。
3. **添加 `dry-run` 模式** — 对破坏性操作（重命名、删除），先用 `--dry-run` 预览结果。
4. **输出日志，不要 print** — 生产脚本使用 `logging` 模块，控制级别和输出格式。
5. **配置外部化** — 邮箱密码、文件路径等不要硬编码，使用环境变量或 `.env` 文件。
6. **幂等性设计** — 脚本运行多次应该得到相同结果，不会产生重复副作用。

自动化脚本的价值在于把重复性劳动压缩到一次编写、长期受益。从小任务开始（批量改文件名），逐步构建更复杂的 pipeline（报表生成 + 邮件发送 + 定时触发），你的工作效率会有质的飞跃。
