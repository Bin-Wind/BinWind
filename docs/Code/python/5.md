# 面向对象编程基础

面向对象编程（OOP）是一种将数据和操作封装在一起的编程范式。Python 的 class 机制直观灵活，本节覆盖类定义、继承、多态、property 装饰器和魔术方法。

## 定义类与实例化

```python
class Dog:
    """一个简单的狗类"""

    # 类属性 - 所有实例共享
    species = "Canis familiaris"

    # 初始化方法（构造函数）
    def __init__(self, name, age):
        self.name = name     # 实例属性
        self.age = age

    # 实例方法
    def bark(self):
        return f"{self.name} 说：汪汪！"

    def info(self):
        return f"{self.name}，{self.age}岁，品种：{self.species}"

# 创建实例
buddy = Dog("Buddy", 3)
max = Dog("Max", 5)

print(buddy.bark())                # Buddy 说：汪汪！
print(max.info())                  # Max，5岁，品种：Canis familiaris
print(buddy.species)               # Canis familiaris

# self 指向调用方法的实例本身 — 这是 Python OOP 的核心约定
```

## 继承 Inheritance

```python
class Animal:
    def __init__(self, name):
        self.name = name

    def speak(self):
        return "..."

class Cat(Animal):
    def speak(self):
        return f"{self.name} 说：喵喵！"

class Bird(Animal):
    def __init__(self, name, can_fly=True):
        super().__init__(name)           # 调用父类 __init__
        self.can_fly = can_fly

    def speak(self):
        return f"{self.name} 说：叽叽！"

# 多态 - 同一接口，不同行为
animals = [Cat("咪咪"), Bird("波利")]
for animal in animals:
    print(animal.speak())
# 咪咪 说：喵喵！
# 波利 说：叽叽！

# 检查继承关系
print(isinstance(Cat("咪咪"), Animal))   # True
print(issubclass(Cat, Animal))           # True
```

## 封装与访问控制

Python 没有真正的 private 属性，而是通过命名约定实现封装。

```python
class BankAccount:
    def __init__(self, owner, balance=0):
        self.owner = owner               # 公开属性
        self._protected = "可访问但不建议"  # 受保护（约定）
        self.__private = balance          # 私有（名称改写）

    def deposit(self, amount):
        if amount > 0:
            self.__private += amount
            return True
        return False

    def get_balance(self):
        return self.__private

account = BankAccount("Alice", 1000)
account.deposit(500)
print(account.get_balance())           # 1500

# print(account.__private)             # AttributeError
# print(account._BankAccount__private) # 1500 — 名称改写后仍可访问
```

## @property 装饰器 - 优雅的 getter/setter

```python
class Temperature:
    def __init__(self, celsius=0):
        self._celsius = celsius

    @property
    def celsius(self):
        """获取摄氏度"""
        return self._celsius

    @celsius.setter
    def celsius(self, value):
        """设置摄氏度，加验证"""
        if value < -273.15:
            raise ValueError("温度不能低于绝对零度")
        self._celsius = value

    @property
    def fahrenheit(self):
        """通过计算属性获得华氏度（只读）"""
        return self._celsius * 9 / 5 + 32

t = Temperature(37)
print(t.celsius)       # 37    - 无需括号，像访问属性一样
print(t.fahrenheit)    # 98.6
t.celsius = 100        # 调用 setter
# t.fahrenheit = 200   # AttributeError - 只读
```

## 常见的 Dunder（双下划线）方法

```python
class Vector:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def __str__(self):
        """print() 和 str() 时的友好显示"""
        return f"Vector({self.x}, {self.y})"

    def __repr__(self):
        """开发者调试用的表示，追求 eval(repr(obj)) == obj"""
        return f"Vector({self.x}, {self.y})"

    def __add__(self, other):
        """重载 + 运算符"""
        return Vector(self.x + other.x, self.y + other.y)

    def __eq__(self, other):
        """重载 == 运算符"""
        if not isinstance(other, Vector):
            return NotImplemented
        return self.x == other.x and self.y == other.y

    def __len__(self):
        """支持 len()"""
        return 2

    def __getitem__(self, index):
        """支持索引访问 v[0], v[1]"""
        return (self.x, self.y)[index]

    def __bool__(self):
        """bool() 判断：零向量视为 False"""
        return self.x != 0 or self.y != 0

v1 = Vector(3, 4)
v2 = Vector(1, 2)
print(v1)                  # Vector(3, 4)
print(v1 + v2)             # Vector(4, 6)
print(v1 == Vector(3, 4))  # True
print(v1[0], v1[1])        # 3 4
print(bool(Vector(0, 0)))  # False
```

## 类方法与静态方法

```python
class DateUtils:
    date_format = "%Y-%m-%d"

    @classmethod
    def set_format(cls, new_format):
        """类方法：操作类属性，常用于工厂模式"""
        cls.date_format = new_format

    @classmethod
    def from_string(cls, date_str):
        """工厂方法：从字符串创建实例"""
        import datetime
        return datetime.datetime.strptime(date_str, cls.date_format)

    @staticmethod
    def is_valid_year(year):
        """静态方法：与类和实例无关的工具函数"""
        return isinstance(year, int) and year > 0

dt = DateUtils.from_string("2024-06-15")
print(DateUtils.is_valid_year(2024))   # True
```

> OOP 适合对现实世界的实体建模。Python 的 OOP 哲学是"我们都是成年人"——约定优于强制，相信程序员会正确使用接口。善用 `@property` 和 dunder 方法，可以让自定义类型像内置类型一样自然。
