# 文件读写与路径操作

文件 I/O 是脚本和数据处理中最常见的操作。Python 提供了直观的 `open` 函数和强大的 `pathlib` 模块，让文件读写和路径操作变得轻松高效。

## 基本文件读写

```python
# 写文件
with open("example.txt", "w", encoding="utf-8") as f:
    f.write("第一行\n")
    f.write("第二行\n")

# 读文件 - 一次性读取
with open("example.txt", "r", encoding="utf-8") as f:
    content = f.read()
    print(content)

# 逐行读取（适合大文件）
with open("example.txt", "r", encoding="utf-8") as f:
    for line in f:
        print(line.strip())
```

## 文件打开模式

| 模式 | 说明 |
|------|------|
| `"r"` | 只读（默认），文件必须存在 |
| `"w"` | 只写，覆盖已有内容，不存在则创建 |
| `"a"` | 追加写入，不存在则创建 |
| `"x"` | 排他创建，文件已存在则失败 |
| `"r+"` | 读写，文件必须存在 |
| `"b"` | 二进制模式，如 `"rb"`、`"wb"` |
| `"t"` | 文本模式（默认），如 `"rt"` |

```python
# 追加写入
with open("log.txt", "a", encoding="utf-8") as f:
    import datetime
    f.write(f"[{datetime.datetime.now()}] 日志条目\n")

# 二进制读写（图片、音频等）
with open("source.png", "rb") as src:
    data = src.read()

with open("copy.png", "wb") as dst:
    dst.write(data)
```

## with 语句 - 上下文管理器

`with` 保证文件在使用后自动关闭，即使在异常发生时也是如此：

```python
# with 语句 = try/finally 的优雅替代
# 等价于：
# f = open("file.txt", "r")
# try:
#     content = f.read()
# finally:
#     f.close()

# 同时打开多个文件
with open("input.txt", "r") as f_in, open("output.txt", "w") as f_out:
    for line in f_in:
        f_out.write(line.upper())
```

## CSV 文件读写

```python
import csv

# 写 CSV
with open("users.csv", "w", newline="", encoding="utf-8") as f:
    writer = csv.writer(f)
    writer.writerow(["姓名", "年龄", "城市"])
    writer.writerows([
        ["张三", 28, "北京"],
        ["李四", 35, "上海"],
        ["王五", 22, "广州"],
    ])

# 读 CSV
with open("users.csv", "r", encoding="utf-8") as f:
    reader = csv.DictReader(f)
    for row in reader:
        print(f"{row['姓名']}，{row['年龄']}岁，住在{row['城市']}")

# 自定义分隔符和引用
with open("data.tsv", "w", newline="", encoding="utf-8") as f:
    writer = csv.writer(f, delimiter="\t", quotechar='"')
    writer.writerow(["A", "B", "C"])
```

## JSON 文件读写

```python
import json

data = {
    "users": [
        {"name": "张三", "age": 28},
        {"name": "李四", "age": 35},
    ],
    "count": 2
}

# 写 JSON（带缩进可读性更好）
with open("data.json", "w", encoding="utf-8") as f:
    json.dump(data, f, ensure_ascii=False, indent=2)

# 读 JSON
with open("data.json", "r", encoding="utf-8") as f:
    loaded = json.load(f)
    print(loaded["users"][0]["name"])         # 张三

# JSON 字符串 <-> Python 对象
json_str = json.dumps(data, ensure_ascii=False)
parsed = json.loads(json_str)
```

## pathlib - 现代化路径操作

pathlib 提供面向对象的路径处理，比 `os.path` 更直观：

```python
from pathlib import Path

# 创建路径对象
root = Path(".")                             # 当前目录
data_dir = Path("data") / "2024" / "raw"     # 用 / 拼接路径

# 创建目录
data_dir.mkdir(parents=True, exist_ok=True)

# 遍历文件
for file in root.glob("*.py"):
    print(file)

for file in root.rglob("*.txt"):              # 递归 glob
    print(file)

# 路径属性
file_path = Path("data/config.json")
print(file_path.name)          # config.json
print(file_path.stem)          # config
print(file_path.suffix)        # .json
print(file_path.parent)        # data
print(file_path.exists())      # True / False

# 读写快捷方法
content = Path("example.txt").read_text(encoding="utf-8")
Path("output.txt").write_text("Hello, Pathlib!", encoding="utf-8")

# 检查路径类型
path = Path("some_path")
if path.is_file():
    print("是文件")
elif path.is_dir():
    print("是目录")

# 获取绝对路径
abs_path = path.resolve()
```

## 常见文件操作示例

```python
from pathlib import Path
import shutil
import os

# 复制文件
shutil.copy("source.txt", "dest.txt")

# 移动/重命名文件
Path("old_name.txt").rename("new_name.txt")
# 或
shutil.move("old_name.txt", "new_name.txt")

# 删除文件
Path("temp.txt").unlink(missing_ok=True)     # missing_ok=True 避免文件不存在报错

# 删除目录树
shutil.rmtree("temp_dir", ignore_errors=True)

# 文件大小
size = Path("large_file.bin").stat().st_size
print(f"{size} bytes")

# 临时文件
import tempfile
with tempfile.NamedTemporaryFile(mode="w", suffix=".txt", delete=False) as tmp:
    tmp.write("临时内容")
    print(f"临时文件路径：{tmp.name}")
```

## 实战：批量重命名脚本

```python
from pathlib import Path

def batch_rename(directory, prefix, extension):
    """将目录下所有指定扩展名的文件重命名为 prefix_序号.ext"""
    files = list(Path(directory).glob(f"*{extension}"))
    for i, file in enumerate(files, start=1):
        new_name = file.parent / f"{prefix}_{i:03d}{extension}"
        file.rename(new_name)
        print(f"{file.name} -> {new_name.name}")

# 使用示例：batch_rename("./photos", "img", ".jpg")
```

> 始终使用 `with` 语句操作文件——它保证资源正确释放。在 Python 3.6+ 项目中，pathlib 应为首选路径工具，用 `Path / "subdir"` 拼接比字符串拼接更清晰、更安全。
