# 装饰器（Decorator）完全指南

装饰器是 Python 最强大的特性之一——它在不修改原函数代码的前提下为函数添加额外行为。理解装饰器意味着真正理解了 Python 中的函数是一等公民。

## 函数是一等公民

在理解装饰器之前，先回顾几个关键概念：

```python
# 函数可以赋值给变量
def greet(name):
    return f"你好, {name}!"

say_hello = greet
print(say_hello("小明"))             # 你好, 小明!

# 函数可以作为参数传递
def execute(func, arg):
    return func(arg)

print(execute(greet, "小红"))        # 你好, 小红!

# 函数可定义在另一个函数内部，并作为返回值
def make_multiplier(factor):
    def multiplier(x):
        return x * factor
    return multiplier

double = make_multiplier(2)
triple = make_multiplier(3)
print(double(5))                     # 10
print(triple(5))                     # 15
```

这些特性构成了闭包（closure）——内部函数记住了其外部作用域中的变量。

## 基本装饰器

```python
import time

# 装饰器就是一个接受函数、返回函数的函数
def timer(func):
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f"{func.__name__} 执行耗时：{elapsed:.4f} 秒")
        return result
    return wrapper

# 使用装饰器
@timer
def slow_function():
    time.sleep(1.2)
    return "完成"

print(slow_function())
# slow_function 执行耗时：1.2003 秒
# 完成
```

`@timer` 等价于 `slow_function = timer(slow_function)`。

## 使用 functools.wraps 保留元信息

```python
import functools

def bad_decorator(func):
    def wrapper(*args, **kwargs):
        """wrapper 的文档"""
        return func(*args, **kwargs)
    return wrapper

def good_decorator(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        """wrapper 的文档"""
        return func(*args, **kwargs)
    return wrapper

@bad_decorator
def foo():
    """foo 的文档"""
    pass

@good_decorator
def bar():
    """bar 的文档"""
    pass

print(foo.__name__)    # wrapper     ← 丢失了
print(foo.__doc__)     # wrapper 的文档

print(bar.__name__)    # bar         ← 正确
print(bar.__doc__)     # bar 的文档
```

**任何时候写装饰器都要用 `@functools.wraps`**，否则函数名、文档字符串、签名都会丢失。

## 带参数的装饰器

装饰器本身也可以接受参数——这需要额外一层嵌套：

```python
import functools
import time

def retry(max_attempts=3, delay=1):
    """重试装饰器：执行失败时自动重试"""

    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            last_exception = None
            for attempt in range(1, max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    last_exception = e
                    print(f"尝试 {attempt}/{max_attempts} 失败：{e}")
                    if attempt < max_attempts:
                        time.sleep(delay)
            raise last_exception
        return wrapper

    return decorator

# 使用带参数的装饰器
@retry(max_attempts=4, delay=0.5)
def unstable_network_call():
    import random
    if random.random() < 0.7:
        raise ConnectionError("网络超时")
    return "成功"

# 语法糖等价于：
# unstable_network_call = retry(max_attempts=4, delay=0.5)(unstable_network_call)
```

## 常用装饰器实战

```python
import functools
import time

# 1. 缓存装饰器
def memoize(func):
    """缓存函数调用结果"""
    cache = {}

    @functools.wraps(func)
    def wrapper(*args):
        if args not in cache:
            cache[args] = func(*args)
        return cache[args]

    return wrapper

@memoize
def fibonacci(n):
    if n < 2:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

print(fibonacci(100))   # 不带缓存几乎算不出来

# 2. 权限检查装饰器
def require_role(role):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(user, *args, **kwargs):
            if user.get("role") != role:
                raise PermissionError(f"需要 {role} 权限")
            return func(user, *args, **kwargs)
        return wrapper
    return decorator

@require_role("admin")
def delete_user(user, target_id):
    return f"用户 {target_id} 已被删除"

# print(delete_user({"role": "user"}, 123))    # PermissionError
print(delete_user({"role": "admin"}, 123))     # 用户 123 已被删除

# 3. 计时 + 日志组合装饰器
def log_and_time(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        print(f"[LOG] 调用 {func.__name__}(args={args}, kwargs={kwargs})")
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f"[LOG] {func.__name__} 返回 {result}, 耗时 {elapsed:.4f}s")
        return result
    return wrapper
```

## 多个装饰器叠加

```python
@timer
@log_and_time
def process_data(x):
    return x * 2

# 执行顺序：从下往上应用（离函数越近的装饰器越先执行）
# 等价于：process_data = timer(log_and_time(process_data))
```

## 类装饰器

```python
class CountCalls:
    """用类实现的装饰器：统计函数调用次数"""

    def __init__(self, func):
        functools.update_wrapper(self, func)
        self.func = func
        self.count = 0

    def __call__(self, *args, **kwargs):
        self.count += 1
        print(f"{self.func.__name__} 已被调用 {self.count} 次")
        return self.func(*args, **kwargs)

@CountCalls
def say_hello():
    return "Hello!"

say_hello()    # say_hello 已被调用 1 次
say_hello()    # say_hello 已被调用 2 次
```

类装饰器的优势是可以通过实例属性保持状态。

## 装饰器在标准库中的应用

```python
# @staticmethod / @classmethod / @property — 最常见的装饰器
# @functools.lru_cache — 内置缓存装饰器
from functools import lru_cache

@lru_cache(maxsize=128)
def expensive_query(id):
    print(f"执行查询 id={id}")
    return id * 10

# @dataclass — 自动生成 __init__、__repr__ 等
from dataclasses import dataclass

@dataclass
class Point:
    x: float
    y: float
```

> 装饰器的核心思想："组合优于修改"。当你需要为多个函数添加相同的前置/后置逻辑时，装饰器是最优雅的方案。写好装饰器的关键是习惯那层"函数套函数"的包裹模式。
