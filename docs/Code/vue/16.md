# 插件开发

Vue 插件（Plugin）通过 `app.use()` 安装，可添加全局组件、指令、provide 数据和全局方法。插件是一个带有 `install` 方法的对象。

## install 函数

```typescript
// plugins/myPlugin.ts
import type { App } from 'vue'

interface Options { prefix?: string }

export default {
  install(app: App, options: Options = {}) {
    const prefix = options.prefix ?? 'Global'
    app.provide('pluginPrefix', prefix)
  }
}

// main.ts
app.use(myPlugin, { prefix: 'MyApp' })
```

## 四大能力

### 1. 注册全局组件

```typescript
import type { App } from 'vue'
import MyButton from '@/components/MyButton.vue'

export default {
  install(app: App) {
    app.component('MyButton', MyButton)
  }
}
```

### 2. 注册全局指令

```typescript
import { vClickOutside, vLazyLoad } from '../directives'

export default {
  install(app: App) {
    app.directive('click-outside', vClickOutside)
    app.directive('lazy-load', vLazyLoad)
  }
}
```

### 3. Provide 全局数据

```typescript
import type { App, InjectionKey } from 'vue'

export const I18N_KEY: InjectionKey<{
  t: (key: string) => string; locale: string
}> = Symbol('i18n')

export default {
  install(app: App, options: { locale: string; messages: Record<string, string> }) {
    app.provide(I18N_KEY, {
      locale: options.locale,
      t: (key: string) => options.messages[key] ?? key
    })
  }
}
// 组件中使用: const i18n = inject(I18N_KEY)!
```

### 4. 扩展全局属性

```typescript
import axios from 'axios'

declare module 'vue' {
  interface ComponentCustomProperties { $http: typeof axios }
}

export default {
  install(app: App) {
    app.config.globalProperties.$http = axios
  }
}
```

## Toast 通知插件

```typescript
import { createApp, h, type App } from 'vue'
import ToastComponent from '@/components/Toast.vue'

export default {
  install(app: App) {
    const el = document.createElement('div')
    document.body.appendChild(el)
    const vm = createApp(ToastComponent).mount(el)
    app.provide('toast', {
      show(msg: string, type = 'info', duration = 3000) {
        ;(vm as any).add({ message: msg, type, duration })
      }
    })
  }
}
```

## 发布 npm 包

**1. 结构：** `src/index.ts` 导出 install，`dist/` 为构建产物。

**2. package.json：**

```json
{
  "main": "./dist/my-plugin.umd.js",
  "module": "./dist/my-plugin.mjs",
  "types": "./dist/index.d.ts",
  "files": ["dist"],
  "peerDependencies": { "vue": "^3.3.0" }
}
```

**3. 构建：**

```typescript
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
import dts from 'vite-plugin-dts'

export default defineConfig({
  plugins: [vue(), dts()],
  build: {
    lib: { entry: 'src/index.ts', name: 'MyPlugin', formats: ['es', 'umd'] },
    rollupOptions: { external: ['vue'], output: { globals: { vue: 'Vue' } } }
  }
})
```

**4. 发布：** `npm login && npm publish --access public`

## 总结

插件核心：安装时注册资源、provide 数据、扩展原型。发布时确保 peerDependencies、ESM/UMD 双格式、TS 类型声明。
