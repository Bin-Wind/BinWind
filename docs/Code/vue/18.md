# HTTP 请求管理

合理的 HTTP 管理不止于调用库。封装应包括拦截器、请求取消、错误统一处理，并以 Composable 暴露给组件。

## Axios 实例与拦截器

```typescript
// api/request.ts
import axios, { AxiosError } from 'axios'
import { useUserStore } from '@/stores/user'
import { ElMessage } from 'element-plus'

const http = axios.create({
  baseURL: import.meta.env.VITE_API_BASE,
  timeout: 10000
})

// 请求拦截 —— Token 注入
http.interceptors.request.use((config) => {
  const token = useUserStore().token
  if (token) config.headers.Authorization = `Bearer ${token}`
  return config
})

// 响应拦截 —— 统一处理
http.interceptors.response.use(
  (res) => {
    const { code, data, message } = res.data
    if (code === 0) return data
    ElMessage.error(message || '请求失败')
    return Promise.reject(new Error(message))
  },
  (error: AxiosError) => {
    if (error.response?.status === 401) {
      useUserStore().logout()
      window.location.href = '/login'
    } else if (error.code === 'ECONNABORTED') {
      ElMessage.error('请求超时')
    } else {
      ElMessage.error('网络错误')
    }
    return Promise.reject(error)
  }
)

export default http
```

## 请求取消

场景：用户快速切换页面时，取消未完成的请求。

```typescript
// composables/useCancelableRequest.ts
import { onUnmounted, ref } from 'vue'

export function useCancelableRequest() {
  const ctrls = ref<AbortController[]>([])

  function createSignal() {
    const c = new AbortController()
    ctrls.value.push(c)
    return c.signal
  }

  function cancelAll() {
    ctrls.value.forEach(c => c.abort())
    ctrls.value = []
  }

  onUnmounted(cancelAll)
  return { createSignal, cancelAll }
}

// 使用
watch(keyword, async (kw) => {
  cancelAll()
  results.value = await searchUsers(kw, createSignal())
})
```

## Composable 封装

```typescript
// composables/useRequest.ts
import { ref, type Ref } from 'vue'

export function useRequest<T>(fetcher: (...args: any[]) => Promise<T>) {
  const data: Ref<T | null> = ref(null)
  const loading = ref(false)
  const error = ref<string | null>(null)

  async function execute(...args: any[]) {
    loading.value = true; error.value = null
    try { data.value = await fetcher(...args) }
    catch (e: any) { error.value = e.message || 'Unknown error' }
    finally { loading.value = false }
  }

  return { data, loading, error, execute }
}
```

## API 层组织

```
api/
├── request.ts          # 实例 + 拦截器
├── modules/
│   ├── user.ts         # /user/*
│   └── article.ts      # /article/*
└── index.ts
```

```typescript
// api/modules/user.ts
import http from '../request'

export interface LoginParams { username: string; password: string }
export interface UserInfo { id: number; name: string; avatar: string }

export const login = (p: LoginParams) => http.post<{ token: string }>('/user/login', p)
export const getUserInfo = () => http.get<UserInfo>('/user/info')
```

## 总结

| 层面 | 职责 |
|------|------|
| request.ts | 实例、拦截器、错误统一处理 |
| modules/ | 按模块分组 API，定义参数与返回类型 |
| composables/ | 封装 loading/error/取消等通用逻辑 |
| 组件 | 调用 composable，专注 UI |
